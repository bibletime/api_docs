32 bool isUtf8(const char * buf) {
33  int i, n;
34  unsigned char c;
35  bool gotone = false;
36 
37  /* Marks characters which never appear in text: */
38  static constexpr unsigned char const F = 0;
39 
40  /* Marks characters which appear in plain ASCII texts: */
41  static constexpr unsigned char const T = 1;
42 
43  /* Marks characters which appear in ISO-8859 texts: */
44  static constexpr unsigned char const I = 2;
45 
46  /* Marks characters which appear in non-ISO extended ASCII (Mac, IBM PC): */
47  static constexpr unsigned char const X = 3;
48 
49  static const unsigned char text_chars[] = {
50  /* BEL BS HT LF FF CR */
51  F, F, F, F, F, F, F, T, T, T, T, F, T, T, F, F, /* 0x0X */
52  /* ESC */
53  F, F, F, F, F, F, F, F, F, F, F, T, F, F, F, F, /* 0x1X */
54  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x2X */
55  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x3X */
56  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x4X */
57  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x5X */
58  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x6X */
59  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, F, /* 0x7X */
60  /* NEL */
61  X, X, X, X, X, T, X, X, X, X, X, X, X, X, X, X, /* 0x8X */
62  X, X, X, X, X, X, X, X, X, X, X, X, X, X, X, X, /* 0x9X */
63  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xaX */
64  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xbX */
65  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xcX */
66  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xdX */
67  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xeX */
68  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I /* 0xfX */
69  };
70  static_assert(std::extent_v<decltype(text_chars)> == 256u);
71 
72  /* *ulen = 0; */
73 
74  for (i = 0; (c = static_cast<unsigned char>(buf[i])); i++) {
75  if ((c & 0x80) == 0) { /* 0xxxxxxx is plain ASCII */
76  /*
77  * Even if the whole file is valid UTF-8 sequences,
78  * still reject it if it uses weird control characters.
79  */
80 
81  if (text_chars[c] != T)
82  return false;
83 
84  }
85  else if ((c & 0x40) == 0) { /* 10xxxxxx never 1st byte */
86  return false;
87  }
88  else { /* 11xxxxxx begins UTF-8 */
89  int following;
90 
91  if ((c & 0x20) == 0) { /* 110xxxxx */
92  following = 1;
93  }
94  else if ((c & 0x10) == 0) { /* 1110xxxx */
95  following = 2;
96  }
97  else if ((c & 0x08) == 0) { /* 11110xxx */
98  following = 3;
99  }
100  else if ((c & 0x04) == 0) { /* 111110xx */
101  following = 4;
102  }
103  else if ((c & 0x02) == 0) { /* 1111110x */
104  following = 5;
105  }
106  else
107  return false;
108 
109  for (n = 0; n < following; n++) {
110  i++;
111 
112  if (!(c = static_cast<unsigned char>(buf[i])))
113  goto done;
114 
115  if ((c & 0x80) == 0 || (c & 0x40))
116  return false;
117  }
118 
119  gotone = true;
120  }
121  }
122 
123 done:
124  return gotone; /* don't claim it's UTF-8 if it's all 7-bit */
125 }
