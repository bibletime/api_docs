51 CSearchAnalysisScene::CSearchAnalysisScene(
52  QString searchedText,
53  CSwordModuleSearch::Results const & results,
54  QObject * parent)
55  : QGraphicsScene(parent)
56  , m_searchedText(std::move(searchedText))
57 {
58  setBackgroundBrush(QBrush(Qt::white));
59  setSceneRect(0, 0, 1, 1);
60 
61  for (auto const & result : results)
62  if ((result.module->type() == CSwordModuleInfo::Bible)
63  || (result.module->type() == CSwordModuleInfo::Commentary))
64  m_results.emplace_back(result);
65 
66  auto const numberOfModules = m_results.size();
67  if (!numberOfModules)
68  return;
69  m_legend = std::make_unique<CSearchAnalysisLegendItem>(&m_results);
70  addItem(m_legend.get());
71  m_legend->setRect(LEFT_BORDER, UPPER_BORDER,
72  LEGEND_WIDTH, LEGEND_INNER_BORDER*2 + ITEM_TEXT_SIZE*numberOfModules + LEGEND_DELTAY*(numberOfModules - 1) );
73  m_legend->show();
74 
75  int moduleIndex = 0;
76  for (auto const & result : m_results) {
77  qApp->processEvents(QEventLoop::AllEvents);
78  for (auto const & moduleresultPtr : result.results) {
79  /* m_results only contains results from Bibles and
80  Commentaries, as filtered above. */
81  BT_ASSERT(dynamic_cast<sword::VerseKey const *>(
82  moduleresultPtr.get()));
83  auto const * const vk =
84  static_cast<sword::VerseKey const *>(
85  moduleresultPtr.get());
86  auto key = std::tuple(vk->getTestament(), vk->getBook());
87  CSearchAnalysisItem * analysisItem;
88  static_assert(std::is_same_v<decltype(key),
89  decltype(m_itemList)::key_type>, "");
90  if (auto const it = m_itemList.find(key); it != m_itemList.end()) {
91  analysisItem = it->second;
92  } else {
93  analysisItem =
94  new CSearchAnalysisItem(
95  QString::fromUtf8(vk->getBookName()),
96  m_results.size());
97  m_itemList.emplace(std::move(key), analysisItem);
98  }
99  auto const count = ++analysisItem->counts()[moduleIndex];
100  m_maxCount = std::max(m_maxCount, count);
101  }
102 
103  ++moduleIndex;
104  }
105 
106  int xPos = static_cast<int>(LEFT_BORDER
107  + m_legend->rect().width()
108  + SPACE_BETWEEN_PARTS);
109  for (auto const & vp : m_itemList) {
110  auto & analysisItem = *vp.second;
111  addItem(&analysisItem);
112  analysisItem.setRect(xPos,
113  UPPER_BORDER,
114  analysisItem.rect().width(),
115  analysisItem.rect().height());
116  xPos += static_cast<int>(analysisItem.width() + SPACE_BETWEEN_PARTS);
117 
118  QString toolTip("<center><b>");
119  toolTip.append(analysisItem.bookName().toHtmlEscaped())
120  .append("</b></center><hr/><table cellspacing=\"0\" "
121  "cellpadding=\"3\" width=\"100%\" height=\"100%\" "
122  "align=\"center\">");
123 
124  int i = 0;
125  for (auto const & result : m_results) {
126  auto const * const info = result.module;
127 
128  auto const count = result.results.size();
129  double const percent =
130  (info && count)
131  ? ((static_cast<double>(analysisItem.counts()[i])
132  * static_cast<double>(100.0))
133  / static_cast<double>(count))
134  : 0.0;
135  toolTip.append("<tr bgcolor=\"white\"><td><b><font color=\"")
136  .append(getColor(i).name()).append("\">")
137  .append(info ? info->name() : QString())
138  .append("</font></b></td><td>")
139  .append(QString::number(analysisItem.counts()[i]))
140  .append(" (")
141  .append(QString::number(percent, 'g', 2))
142  .append("%)</td></tr>");
143  ++i;
144  }
145  toolTip.append("</table>");
146  analysisItem.setToolTip(toolTip);
147  }
148  setSceneRect(0, 0, xPos + BAR_WIDTH + (m_results.size() - 1)*BAR_DELTAX + RIGHT_BORDER, height() );
149  slotResized();
150 }

152 /** No descriptions */
153 void CSearchAnalysisScene::slotResized() {
154  double scaleFactor;
155  if (m_maxCount <= 0) {
156  scaleFactor = 0.0;
157  } else {
158  scaleFactor = static_cast<double>(height() - UPPER_BORDER - LOWER_BORDER - BAR_LOWER_BORDER - 100 - (m_results.size() - 1) * BAR_DELTAY)
159  / static_cast<double>(m_maxCount);
160  }
161  for (auto const & vp : m_itemList) {
162  auto & analysisItem = *vp.second;
163  analysisItem.setScaleFactor(scaleFactor);
164  analysisItem.setRect(analysisItem.rect().x(),
165  UPPER_BORDER,
166  BAR_WIDTH + (m_results.size() - 1) * BAR_DELTAX,
167  height() - LOWER_BORDER - BAR_LOWER_BORDER);
168  }
169  update();
170 }
