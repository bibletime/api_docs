70 /** Sets the module for this key */
71 void CSwordVerseKey::setModule(const CSwordModuleInfo *newModule) {
72  BT_ASSERT(newModule);
73  if (m_module == newModule) return;
74  BT_ASSERT(newModule->type() == CSwordModuleInfo::Bible ||
75  newModule->type() == CSwordModuleInfo::Commentary);
76 
77  CSwordBibleModuleInfo const * bible = static_cast<CSwordBibleModuleInfo const *>(newModule);
78  const char * newVersification =
79  static_cast<sword::VerseKey *>(
80  bible->swordModule().getKey())->getVersificationSystem();
81  bool inVersification = true;
82 
83  if (strcmp(m_key.getVersificationSystem(), newVersification)) {
84  /// Remap key position to new versification
85  sword::VerseKey oldKey(m_key);
86 
87  m_key.setVersificationSystem(newVersification);
88 
89  m_key.positionFrom(oldKey);
90  inVersification = !m_key.popError();
91  }
92 
93  m_module = newModule;
94 
95  emitAfterChanged();
96 
97  if(inVersification) {
98  /// Limit to Bible bounds
99  if (m_key._compare(bible->lowerBound().m_key) < 0) {
100  setKey(bible->lowerBound().m_key);
101  }
102  if (m_key._compare(bible->upperBound().m_key) > 0) {
103  setKey(bible->upperBound().m_key);
104  }
105  }
106 
107  m_valid = inVersification;
108 }

123 QString CSwordVerseKey::bookName() const {
124  using CSBMI = CSwordBibleModuleInfo;
125  int min = 0;
126  int max = 1;
127 
128  const CSBMI *bible = dynamic_cast<const CSBMI*>(module());
129  if (bible != nullptr) {
130  const bool hasOT = bible->hasOldTestament();
131  const bool hasNT = bible->hasNewTestament();
132 
133  if (hasOT && hasNT) {
134  min = 0;
135  max = 1;
136  }
137  else if (hasOT && !hasNT) {
138  min = 0;
139  max = 0;
140  }
141  else if (!hasOT && hasNT) {
142  min = 1;
143  max = 1;
144  }
145  else if (!hasOT && !hasNT) {
146  min = 0;
147  max = -1; //no loop
148  }
149  }
150 
151  if ((m_key.getTestament() >= min + 1) && (m_key.getTestament() <= max + 1) && (m_key.getBook() <= m_key.BMAX[min])) {
152  return QString::fromUtf8(m_key.getBookName());
153  }
154 
155  //return QString::fromUtf8( books[min][0].name ); //return the first book, i.e. Genesis
156  return QString();
157 }

201 bool CSwordVerseKey::next( const JumpType type ) {
202  using CSBMI = CSwordBibleModuleInfo;
203 
204  m_key.popError(); //clear Error status
205  bool ret = true;
206 
207  switch (type) {
208 
209  case UseBook: {
210  const int currentTestament = m_key.getTestament();
211  const int currentBook = m_key.getBook();
212 
213  if ((currentTestament == 2) && (currentBook >= m_key.BMAX[currentTestament-1])) { //Revelation, i.e. end of navigation
214  return false;
215  }
216  else if ((currentTestament == 1) && (currentBook >= m_key.BMAX[currentTestament-1])) { //Malachi, switch to the NT
217  m_key.setTestament(currentTestament + 1);
218  m_key.setBook(1);
219  }
220  else {
221  m_key.setBook(m_key.getBook() + 1);
222  }
223  break;
224  }
225 
226  case UseChapter: {
227  m_key.setChapter(m_key.getChapter() + 1);
228  break;
229  }
230 
231  case UseVerse: {
232  if (!m_module) {
233  m_key.setVerse(m_key.getVerse() + 1);
234  } else {
235  auto & m = m_module->swordModule();
236  const bool oldStatus = m.isSkipConsecutiveLinks();
237  m.setSkipConsecutiveLinks(true);
238 
239  auto * vKey = static_cast<sword::VerseKey *>(m.getKey());
240 
241  // disable headings for next verse
242  bool const oldHeadingsStatus = vKey->isIntros();
243  vKey->setIntros(true);
244  //don't use setKey(), that would create a new key without Headings set
245  vKey->setText(key().toUtf8().constData());
246 
247  m++;
248 
249  vKey = static_cast<sword::VerseKey *>(m.getKey());
250  vKey->setIntros(oldHeadingsStatus);
251  m.setSkipConsecutiveLinks(oldStatus);
252 
253  if (!m.popError()) {
254  setKey(QString::fromUtf8(vKey->getText()));
255  } else {
256  // Verse(Verse()+1);
257  //don't change the key, restore the module's position
258  vKey->setText(key().toUtf8().constData());
259  ret = false;
260  break;
261  }
262  }
263 
264  break;
265  }
266 
267  default:
268  return false;
269  }
270 
271  const CSBMI *bible = dynamic_cast<const CSBMI*>(module());
272  if (bible != nullptr) {
273  if (m_key._compare(bible->lowerBound().m_key) < 0 ) {
274  setKey(bible->lowerBound().m_key);
275  ret = false;
276  }
277 
278  if (m_key._compare(bible->upperBound().m_key) > 0 ) {
279  setKey(bible->upperBound().m_key);
280  ret = false;
281  }
282 
283  emitAfterChanged();
284  return ret;
285  }
286  else if (m_key.popError()) { //we have no module, so take care of VerseKey::Error()
287  return false;
288  }
289 
290  emitAfterChanged();
291  return ret;
292 }
