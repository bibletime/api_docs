202 bool BibleTime::moduleUnlock(CSwordModuleInfo * module, QWidget * parent) {
203  BT_ASSERT(module);
204 
205  /// \todo Write a proper unlocking dialog with integrated error messages.
206  BtMessageInputDialog unlockKeyInputDialog(
207  tr("Unlock Work"),
208  tr("Enter the unlock key for %1.").arg(module->name()),
209  BtMessageInputDialog::Password,
210  module->config(CSwordModuleInfo::CipherKey),
211  module->getUnlockInfo(),
212  parent);
213 
214  while (unlockKeyInputDialog.exec() == QDialog::Accepted) {
215  module->unlock(unlockKeyInputDialog.getUserInput());
216 
217  /// \todo refactor this module reload
218  /* There is currently a deficiency in SWORD 1.8.1 in that
219  backend->setCipherKey() does not work correctly for modules from
220  which data was already fetched. Therefore we have to reload the
221  modules. */
222  {
223  auto const moduleName(module->name());
224  auto & backend = *CSwordBackend::instance();
225  backend.reloadModules(CSwordBackend::OtherChange);
226  module = backend.findModuleByName(moduleName);
227  BT_ASSERT(module);
228  }
229 
230  // Return true if the module was succesfully unlocked:
231  if (!module->isLocked())
232  return true;
233 
234  message::showWarning(
235  parent,
236  tr("Warning: Invalid unlock key!"),
237  tr("The unlock key you provided did not properly unlock "
238  "this module. Please try again."));
239  }
240  return false;
241 }

262 void BibleTime::processCommandline(bool ignoreSession, const QString &bibleKey) {
263  if (btConfig().value<bool>("state/crashedTwoTimes", false)) {
264  return;
265  }
266 
267  // Restore workspace if not not ignoring session data:
268  if (!ignoreSession)
269  reloadProfile();
270 
271  if (btConfig().value<bool>("state/crashedLastTime", false)) {
272  return;
273  }
274 
275  if (!bibleKey.isNull()) {
276  CSwordModuleInfo* bible = btConfig().getDefaultSwordModuleByType("standardBible");
277  if (bibleKey == "random") {
278  CSwordVerseKey vk(nullptr);
279  auto const newIndex = randInt<decltype(vk.index())>(0, 31100);
280  vk.positionToTop();
281  vk.setIndex(newIndex);
282  createReadDisplayWindow(bible, vk.key());
283  } else {
284  createReadDisplayWindow(bible, bibleKey);
285  }
286 
287  /*
288  We are sure only one window is open - it should be displayed
289  fullscreen in the working area:
290  */
291  m_mdi->myTileVertical();
292  }
293 
294  if (btConfig().value<bool>("state/crashedLastTime", false)) {
295  btConfig().setValue("state/crashedTwoTimes", true);
296  }
297  else {
298  btConfig().setValue("state/crashedLastTime", true);
299  }
300  btConfig().sync();
301 
302  // temporary for testing
303  Q_EMIT colorThemeChanged();
304 }

306 bool BibleTime::event(QEvent* event) {
307  if (event->type() == QEvent::PaletteChange) {
308  Q_EMIT colorThemeChanged();
309  // allow to continue to update other parts of Qt widgets
310  }
311  if (event->type() == QEvent::KeyPress) {
312  QKeyEvent *keyEvent = static_cast<QKeyEvent*>(event);
313  if (keyEvent->modifiers() > 0)
314  return false;
315  int value = keyEvent->key();
316  bool accepted = autoScrollAnyKey(value);
317  if (accepted) {
318  return true;
319  }
320  }
321  return QMainWindow::event(event);
322 }
