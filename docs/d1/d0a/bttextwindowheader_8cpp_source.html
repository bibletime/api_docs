64 void BtTextWindowHeader::adjustWidgetCount(bool adjustToZero) {
65  int widgetCountDifference = 0;
66  if (adjustToZero) {
67  widgetCountDifference = m_widgetList.count();
68  }
69  else {
70  widgetCountDifference = m_widgetList.count() - (m_modules.count());
71  }
72  if (m_moduleType == CSwordModuleInfo::GenericBook && !adjustToZero) {
73  widgetCountDifference = (1 - m_widgetList.count()) * -1;
74  }
75  //if there are more buttons than modules, delete buttons
76  if (widgetCountDifference > 0) {
77  while (widgetCountDifference) {
78  // it should be safe to delete the button later
79  BtTextWindowHeaderWidget* w = m_widgetList.takeFirst();
80  w->setParent(nullptr);
81  w->deleteLater();
82  widgetCountDifference--;
83  }
84  }
85  // if there are more modules than buttons, add buttons
86  if (widgetCountDifference < 0) {
87  while (widgetCountDifference) {
88  addWidget();
89  widgetCountDifference++;
90  }
91  }
92 }
