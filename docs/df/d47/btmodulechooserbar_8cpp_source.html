37void BtModuleChooserBar::adjustButtonCount(int const numButtons,
38 CDisplayWindow * const window)
39{
40 if (m_buttonList.size() > numButtons) {
41 auto toRemove = m_buttonList.size() - numButtons;
42 do {
43 delete m_buttonList.takeFirst();
44 } while (--toRemove);
45 } else if (numButtons > m_buttonList.size()) {
46 auto toAdd = numButtons - m_buttonList.size();
47 do {
48 addButton(window);
49 } while (--toAdd);
50 }
51}

53BtModuleChooserButton *
54BtModuleChooserBar::addButton(CDisplayWindow * const window) {
55 BtModuleChooserButton* b = new BtModuleChooserButton(*m_moduleType, this);
56 QAction* a = addWidget(b);
57 m_buttonList.append(b);
58
59 /* The button sends signals directly to the window which then signals back
60 when the module list has changed. Changes to the module list may mean
61 deletion of the buttons, hence we must queue these signals. Otherwise,
62 when triggered via QAction::triggered(), a mouse release event on the
63 widget with that QAction may follow, but the widget might already have
64 been deleted. */
65 BT_CONNECT(b, &BtModuleChooserButton::sigModuleAdd,
66 window, &CDisplayWindow::slotAddModule,
67 Qt::QueuedConnection);
68 BT_CONNECT(b, &BtModuleChooserButton::sigModuleReplace,
69 window, &CDisplayWindow::slotReplaceModule,
70 Qt::QueuedConnection);
71 BT_CONNECT(b, &BtModuleChooserButton::sigModuleRemove,
72 window, &CDisplayWindow::slotRemoveModule,
73 Qt::QueuedConnection);
74
75 a->setVisible(true);
76 return b;
77}

81void BtModuleChooserBar::associateWithWindow(CDisplayWindow * const window) {
82 auto const & modules = window->modules();
83 m_moduleType = modules.first()->type();
84
85 clear();
86 qDeleteAll(m_buttonList);
87 m_buttonList.clear();
88
89 //if (!useModules.count()) return;
90 for (int i = 0; i < modules.size(); i++) {
91 addButton(window);
92 }
93 if (!(*m_moduleType == CSwordModuleInfo::GenericBook)) {
94 addButton(window); // for ADD button
95 } else {
96 BT_ASSERT(modules.size() == 1);
97 }
98 updateButtonMenus(modules);
99
100 BT_CONNECT(window, &CDisplayWindow::sigModuleListChanged,
101 this, qOverload<BtModuleList>(&BtModuleChooserBar::setModules));
102}
