49 void BtModuleChooserBar::adjustButtonCount(bool adjustToZero) {
50  int buttonCountDifference = 0;
51  if (adjustToZero) {
52  buttonCountDifference = m_buttonList.count();
53  }
54  else {
55  buttonCountDifference = m_buttonList.count() - (m_modules.count() + 1);
56  }
57  if (m_moduleType == CSwordModuleInfo::GenericBook && !adjustToZero) {
58  buttonCountDifference = (1 - m_buttonList.count()) * -1;
59  }
60  //if there are more buttons than modules, delete buttons
61  if (buttonCountDifference > 0) {
62  for (int j = 0; j < buttonCountDifference; j++) {
63  delete m_buttonList.takeFirst();
64  }
65  }
66  // if there are more modules than buttons, add buttons
67  if (buttonCountDifference < 0) {
68  for (int i = (buttonCountDifference * (-1)); i > 0; i--) {
69  addButton();
70  }
71  }
72 }

80 BtModuleChooserButton* BtModuleChooserBar::addButton() {
81  BtModuleChooserButton* b = new BtModuleChooserButton(m_moduleType, this);
82  QAction* a = addWidget(b);
83  m_buttonList.append(b);
84 
85  /* The button sends signals directly to the window which then signals back
86  when the module list has changed. Changes to the module list may mean
87  deletion of the buttons, hence we must queue these signals. Otherwise,
88  when triggered via QAction::triggered(), a mouse release event on the
89  widget with that QAction may follow, but the widget might already have
90  been deleted. */
91  BT_CONNECT(b, &BtModuleChooserButton::sigModuleAdd,
92  m_window, &CDisplayWindow::slotAddModule,
93  Qt::QueuedConnection);
94  BT_CONNECT(b, &BtModuleChooserButton::sigModuleReplace,
95  m_window, &CDisplayWindow::slotReplaceModule,
96  Qt::QueuedConnection);
97  BT_CONNECT(b, &BtModuleChooserButton::sigModuleRemove,
98  m_window, &CDisplayWindow::slotRemoveModule,
99  Qt::QueuedConnection);
100 
101  a->setVisible(true);
102  return b;
103 }
