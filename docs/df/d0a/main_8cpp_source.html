70 /**
71  Parses all command-line arguments.
72  \param[out] ignoreSession Whether --ignore-session was specified.
73  \param[out] openBibleKey Will be set to --open-default-bible if specified.
74  \retval -1 Parsing was successful, the application should exit with
75  EXIT_SUCCESS.
76  \retval 0 Parsing was successful.
77  \retval 1 Parsing failed, the application should exit with EXIT_FAILURE.
78 */
79 int parseCommandLine(bool & showDebugMessages,
80  bool & ignoreSession,
81  QString & openBibleKey)
82 {
83  QStringList args = BibleTimeApp::arguments();
84  for (int i = 1; i < args.size(); i++) {
85  const QString &arg = args.at(i);
86  if (arg == "--help"
87  || arg == "-h"
88  || arg == "/?"
89  || arg == "/h")
90  {
91  printHelp(args.at(0));
92  return -1;
93  } else if (arg == "--version"
94  || arg == "-V"
95  || arg == "/V")
96  {
97  std::cout << "BibleTime " BT_VERSION << std::endl;
98  return -1;
99  } else if (arg == "--debug") {
100  showDebugMessages = true;
101  } else if (arg == "--ignore-session") {
102  ignoreSession = true;
103  } else if (arg == "--open-default-bible") {
104  i++;
105  if (i < args.size()) {
106  openBibleKey = args.at(i);
107  } else {
108  std::cerr << qPrintable(QObject::tr(
109  "Error: %1 expects an argument.")
110  .arg("--open-default-bible")) << ' '
111  << qPrintable(QObject::tr("See --help for details."))
112  << std::endl;
113  return 1;
114  }
115  } else {
116  std::cerr << qPrintable(QObject::tr(
117  "Error: Invalid command-line argument: %1")
118  .arg(arg)) << std::endl;
119  return 1;
120  }
121  }
122  return 0;
123 }
