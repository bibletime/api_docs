92  // Connect signals:
93  BT_CONNECT(m_bookshelfWidget->treeView(), &BtBookshelfView::moduleActivated,
94  [this](CSwordModuleInfo * module) {
95  if (!module->isLocked()) {
96  Q_EMIT moduleOpenTriggered(module);

115  auto const moduleName(module->name());
116  if (BibleTime::moduleUnlock(module)) {
117  // Re-initialize module pointer:
118  auto const & backend = CSwordBackend::instance();

120  BT_ASSERT(module);
121  Q_EMIT moduleOpenTriggered(module);

124  });
125  BT_CONNECT(m_bookshelfWidget->treeView(), &BtBookshelfView::moduleHovered,

144  };
145  BT_CONNECT(bookshelfModelPtr.get(), &BtBookshelfModel::rowsInserted,
146  this/*needed*/, modulesChangedSlot);
147  BT_CONNECT(bookshelfModelPtr.get(), &BtBookshelfModel::rowsRemoved,
148  this/*needed*/, std::move(modulesChangedSlot));
149  BT_CONNECT(m_installButton, &QPushButton::clicked,
150  this, &BtBookshelfDockWidget::installWorksClicked);
151 
152  retranslateUi();

155 void BtBookshelfDockWidget::initMenus() {
156  namespace RM = CResMgr::mainIndex;
157 
158  m_itemContextMenu = new QMenu(this);
159  auto const addMenuAction =
160  [this](void (BtBookshelfDockWidget::* signal)(CSwordModuleInfo *)) {
161  auto * const action = new QAction(this);
162  BT_CONNECT(action, &QAction::triggered,
163  [this, signal] {
164  if (CSwordModuleInfo * const module =
165  static_cast<CSwordModuleInfo *>(
166  m_itemContextMenu->property("BtModule")
167  .value<void *>()))
168  Q_EMIT (this->*signal)(module);
169  });
170  m_itemContextMenu->addAction(action);
171  return action;
172  };
173 
174  m_itemOpenAction =
175  addMenuAction(&BtBookshelfDockWidget::moduleOpenTriggered);
176 
177  m_itemSearchAction =
178  addMenuAction(&BtBookshelfDockWidget::moduleSearchTriggered);
179  m_itemSearchAction->setIcon(RM::search::icon());
180 
181  m_itemUnlockAction =
182  addMenuAction(&BtBookshelfDockWidget::moduleUnlockTriggered);
183  m_itemUnlockAction->setIcon(RM::unlockModule::icon());
184 
185  m_itemAboutAction =
186  addMenuAction(&BtBookshelfDockWidget::moduleAboutTriggered);
187  m_itemAboutAction->setIcon(RM::aboutModule::icon());
188 
189  BT_CONNECT(m_itemContextMenu, &QMenu::aboutToShow,
190  [this]{
191  void * v =
192  m_itemContextMenu->property("BtModule")
193  .value<void*>();
194  CSwordModuleInfo *module = static_cast<CSwordModuleInfo*>(v);
195  m_itemOpenAction->setEnabled(!module->isLocked());
196  m_itemSearchAction->setText(
197  tr("&Search in %1...").arg(module->name()));
198  m_itemSearchAction->setEnabled(!module->isLocked());
199  m_itemUnlockAction->setEnabled(module->isLocked());
200  });
201 }
