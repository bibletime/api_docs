39 bool savePlainFile(const QString & filename,
40  const QString & text,
41  QTextCodec * const fileCodec)
42 {
43  BT_ASSERT(fileCodec);
44  BT_ASSERT(!filename.isEmpty());
45 
46  QFile saveFile(filename);
47 
48  if (saveFile.exists())
49  saveFile.remove();
50 
51  if (saveFile.open(QIODevice::ReadWrite)) {
52  QTextStream textstream(&saveFile);
53  textstream.setCodec(fileCodec);
54  textstream << text;
55  textstream.flush();
56  saveFile.close();
57  if (saveFile.error() == QFile::NoError)
58  return true;
59 
60  QMessageBox::critical(nullptr, QObject::tr("Error"),
61  QString::fromLatin1("<qt>%1<br/><b>%2</b></qt>")
62  .arg(QObject::tr("Error while writing to file."))
63  .arg(QObject::tr("Please check that enough disk space is available.")));
64  }
65  else {
66  QMessageBox::critical(nullptr, QObject::tr("Error"),
67  QString::fromLatin1("<qt>%1<br/><b>%2</b></qt>")
68  .arg(QObject::tr("The file couldn't be opened for saving."))
69  .arg(QObject::tr("Please check permissions etc.")));
70  }
71 
72  return false;
73 }
