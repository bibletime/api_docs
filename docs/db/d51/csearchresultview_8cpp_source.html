151 {
152  clear();
153 
154  if (!m) return;

156  m_module = m;
157  if (result.empty())
158  return;
159 
160  setUpdatesEnabled(false);
161 
162  QTreeWidgetItem* oldItem = nullptr;
163  QTreeWidgetItem* item = nullptr;
164  for (auto const & keyPtr : result) {
165  item = new QTreeWidgetItem(this, oldItem);
166  item->setText(0, QString::fromUtf8(keyPtr->getText()));
167  oldItem = item;
168  }
169
