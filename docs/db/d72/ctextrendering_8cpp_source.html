53 CTextRendering::KeyTreeItem::KeyTreeItem(const QString &startKey,
54  const QString &stopKey,
55  const CSwordModuleInfo *module,
56  const Settings &settings)
57  : m_settings(settings)
58  , m_key(startKey)
59  , m_stopKey(stopKey)
60 {
61  BT_ASSERT(module);
62  m_moduleList.append(module);
63 
64  //use the start and stop key to ceate our child items
65 
66  if (module->type() == CSwordModuleInfo::Bible) {
67  CSwordVerseKey start(module);
68  start.setKey(startKey);
69 
70  CSwordVerseKey stop(module);
71  stop.setKey(stopKey);
72 
73  if (!m_key.isEmpty() && !m_stopKey.isEmpty()) { //we have a range of keys
74  bool ok = true;
75 
76  while (ok && ((start < stop) || (start == stop)) ) { //range
77  m_childList.emplace_back(start.key(),
78  module,
79  KeyTreeItem::Settings{
80  false,
81  settings.keyRenderingFace});
82  ok = start.next(CSwordVerseKey::UseVerse);
83  }
84  }
85  else if (m_key.isEmpty()) {
86  m_childList.emplace_back(startKey,
87  module,
88  KeyTreeItem::Settings{
89  false,
90  settings.keyRenderingFace});
91  }
92  }
93  else if ((module->type() == CSwordModuleInfo::Lexicon) || (module->type() == CSwordModuleInfo::Commentary) ) {
94  m_childList.emplace_back(startKey,
95  module,
96  KeyTreeItem::Settings{
97  false,
98  KeyTreeItem::Settings::NoKey});
99  }
100  else if (module->type() == CSwordModuleInfo::GenericBook) {
101  m_childList.emplace_back(startKey,
102  module,
103  KeyTreeItem::Settings{
104  false,
105  KeyTreeItem::Settings::NoKey});
106  }
107 
108  //make it into "<simple|range> (modulename)"
109 
110  if (startKey == stopKey) {
111  m_alternativeContent = startKey;
112  }
113  else {
114  sword::VerseKey vk(startKey.toUtf8().constData(), stopKey.toUtf8().constData());
115 
116  if (vk.getLowerBound().getBook() != vk.getUpperBound().getBook()) {
117  m_alternativeContent = QString::fromUtf8(vk.getRangeText());
118  }
119  else if (vk.getLowerBound().getChapter() != vk.getUpperBound().getChapter()) {
120  m_alternativeContent = QString("%1 - %2:%3")
121  .arg(QString::fromUtf8(vk.getLowerBound().getText()))
122  .arg(vk.getUpperBound().getChapter())
123  .arg(vk.getUpperBound().getVerse());
124  }
125  else { //only verses differ (same book, same chapter)
126  m_alternativeContent = QString("%1 - %2")
127  .arg(QString::fromUtf8(vk.getLowerBound().getText()))
128  .arg(vk.getUpperBound().getVerse());
129  }
130  }
131 
132  m_alternativeContent.append(" (").append(module->name()).append(")");
133  m_alternativeContent.prepend("<div class=\"rangeheading\" dir=\"ltr\">").append("</div>"); //insert the right tags
134 }

156 const QString CTextRendering::renderKeyTree(const KeyTree &tree) {
157  //CSwordBackend::instance()()->setDisplayOptions( m_displayOptions );
158  CSwordBackend::instance()->setFilterOptions(m_filterOptions);
159 
160  const BtConstModuleList modules = collectModules(tree);
161  QString t;
162 
163  //optimization for entries with the same key
164 
165  if (modules.count() == 1) { //this optimizes the rendering, only one key created for all items
166  std::unique_ptr<CSwordKey> key(
167  CSwordKey::createInstance(modules.first()));
168  for (auto const & item : tree) {
169  key->setKey(item.key());
170  t.append(renderEntry(item, key.get()));
171  }
172  }
173  else {
174  for (auto const & item : tree)
175  t.append(renderEntry(item));
176  }
177 
178  return finishText(t, tree);
179 }

181 const QString CTextRendering::renderKeyRange(
182  CSwordVerseKey const & lowerBound,
183  CSwordVerseKey const & upperBound,
184  const BtConstModuleList &modules,
185  const QString &highlightKey,
186  const KeyTreeItem::Settings &keySettings)
187 {
188 
189  if (lowerBound == upperBound) // same key, render single key:
190  return renderSingleKey(lowerBound.key(), modules, keySettings);
191 
192  // Render range:
193  BT_ASSERT(lowerBound < upperBound);
194  KeyTree tree;
195  KeyTreeItem::Settings settings = keySettings;
196 
197  auto curKey = lowerBound;
198  do {
199  //make sure the key given by highlightKey gets marked as current key
200  settings.highlight = (!highlightKey.isEmpty() ? (curKey.key() == highlightKey) : false);
201 
202  /**
203  \todo We need to take care of linked verses if we render one or
204  (esp) more modules. If the verses 2,3,4,5 are linked to 1,
205  it should be displayed as one entry with the caption 1-5.
206  */
207 
208  if (curKey.chapter() == 0) { // range was 0:0-1:x, render 0:0 first and jump to 1:0
209  curKey.setVerse(0);
210  tree.emplace_back(curKey.key(), modules, settings);
211  curKey.setChapter(1);
212  curKey.setVerse(0);
213  }
214  tree.emplace_back(curKey.key(), modules, settings);
215  if (!curKey.next(CSwordVerseKey::UseVerse)) {
216  /// \todo Notify the user about this failure.
217  break;
218  }
219  } while (curKey < upperBound);
220  return renderKeyTree(tree);
221 }

233 QString CTextRendering::renderEntry(KeyTreeItem const & i, CSwordKey * k)
234 {
235  if (i.hasAlternativeContent()) {
236  QString ret = i.settings().highlight
237  ? "<div class=\"currententry\">"
238  : "<div class=\"entry\">";
239  ret.append(i.getAlternativeContent());
240 
241  if (!i.childList().empty()) {
242  KeyTree const & tree = i.childList();
243 
244  BtConstModuleList const modules(collectModules(tree));
245 
246  if (modules.count() == 1)
247  // insert the direction into the surrounding div:
248  ret.insert(5,
249  QString("dir=\"%1\" ")
250  .arg(modules.first()->textDirectionAsHtml()));
251 
252  for (auto const & item : tree)
253  ret.append(renderEntry(item));
254  }
255 
256  ret.append("</div>");
257  return ret; // WARNING: Return already here!
258  }
259 
260 
261  BtConstModuleList const & modules(i.modules());
262  if (modules.isEmpty())
263  return ""; // no module present for rendering
264 
265  std::unique_ptr<CSwordKey> scoped_key(
266  !k ? CSwordKey::createInstance(modules.first()) : nullptr);
267  CSwordKey * const key = k ? k : scoped_key.get();
268  BT_ASSERT(key);
269 
270  CSwordVerseKey * const myVK = dynamic_cast<CSwordVerseKey *>(key);
271  if (myVK)
272  myVK->setIntros(true);
273 
274  QString renderedText((modules.count() > 1) ? "\n\t\t<tr>\n" : "\n");
275  // Only insert the table stuff if we are displaying parallel.
276 
277  //declarations out of the loop for optimization
278  QString entry;
279  bool isRTL;
280  QString preverseHeading;
281  QString key_renderedText;
282 
283  for (auto const & modulePtr : modules) {
284  BT_ASSERT(modulePtr);
285  if (myVK) {
286  key->setModule(*modules.begin());
287  key->setKey(i.key());
288 
289  // this would change key position due to v11n translation
290  key->setModule(modulePtr);
291  } else {
292  key->setModule(modulePtr);
293  key->setKey(i.key());
294  }
295 
296  // indicate that key was changed
297  i.setMappedKey(key->key() != i.key() ? key : nullptr);
298 
299 
300  isRTL = (modulePtr->textDirection() == CSwordModuleInfo::RightToLeft);
301  entry = QString();
302 
303  auto & swModule = modulePtr->module();
304  auto const langAttr =
305  QString(" xml:lang=\"%1\" lang=\"%1\"").arg(
306  modulePtr->language()->abbrev());
307 
308  if (key->isValid() && i.key() == key->key()) {
309  key_renderedText = key->renderedText();
310 
311  // if key was expanded
312  if (CSwordVerseKey const * const vk =
313  dynamic_cast<CSwordVerseKey *>(key))
314  {
315  if (vk->isBoundSet()) {
316  CSwordVerseKey pk(*vk);
317  auto const lowerBoundIndex = vk->lowerBound().index();
318  auto const upperBoundIndex = vk->upperBound().index();
319  for (auto i = lowerBoundIndex; i < upperBoundIndex; ++i) {
320  key_renderedText += " ";
321  pk.setIndex(i + 1);
322  key_renderedText += pk.renderedText();
323  }
324  }
325  }
326  } else {
327  key_renderedText = "<span class=\"inactive\">&#8212;</span>";
328  }
329 
330  if (m_filterOptions.headings && key->isValid() && i.key() == key->key()) {
331 
332  // only process EntryAttributes, do not render, this might destroy the EntryAttributes again
333  swModule.renderText(nullptr, -1, 0);
334 
335  for (auto const & vp
336  : swModule.getEntryAttributes()["Heading"]["Preverse"])
337  {
338  QString unfiltered(QString::fromUtf8(vp.second.c_str()));
339 
340  /// \todo This is only a preliminary workaround to strip the tags:
341  {
342  static QRegExp const staticFilter(
343  "(.*)<title[^>]*>(.*)</title>(.*)");
344  QRegExp filter(staticFilter);
345  while (filter.indexIn(unfiltered) >= 0)
346  unfiltered = filter.cap(1) + filter.cap(2) + filter.cap(3);
347  }
348 
349  // Filter out offending self-closing div tags, which are bad HTML
350  {
351  static QRegExp const staticFilter("(.*)<div[^>]*/>(.*)");
352  QRegExp filter(staticFilter);
353  while (filter.indexIn(unfiltered) >= 0)
354  unfiltered = filter.cap(1) + filter.cap(2);
355  }
356 
357  preverseHeading = unfiltered;
358 
359  /// \todo Take care of the heading type!
360  if (!preverseHeading.isEmpty()) {
361  entry.append("<div ")
362  .append(langAttr)
363  .append(" class=\"sectiontitle\">")
364  .append(preverseHeading)
365  .append("</div>");
366  }
367  }
368  }
369 
370  entry.append(m_displayOptions.lineBreaks ? "<div class=\"" : "<div class=\"inline ");
371 
372  if (modules.count() == 1) //insert only the class if we're not in a td
373  entry.append( i.settings().highlight ? "currententry " : "entry " );
374  entry.append("\"");
375  entry.append(langAttr).append(isRTL ? " dir=\"rtl\">" : " dir=\"ltr\">");
376 
377  //keys should normally be left-to-right, but this doesn't apply in all cases
378  if(key->isValid() && i.key() == key->key())
379  entry.append("<span class=\"entryname\" dir=\"ltr\">").append(entryLink(i, *modulePtr)).append("</span>");
380 
381  if (m_addText)
382  entry.append(key_renderedText);
383 
384  if (!i.childList().empty())
385  for (auto const & item : i.childList())
386  entry.append(renderEntry(item));
387 
388  entry.append("</div>");
389 
390  if (modules.count() == 1) {
391  renderedText.append("\t\t").append(entry).append("\n");
392  } else {
393  renderedText.append("\t\t<td class=\"")
394  .append(i.settings().highlight ? "currententry" : "entry")
395  .append("\" ")
396  .append(langAttr)
397  .append(" dir=\"")
398  .append(isRTL ? "rtl" : "ltr")
399  .append("\">\n")
400  .append( "\t\t\t" ).append( entry ).append("\n")
401  .append("\t\t</td>\n");
402  }
403  }
404 
405  if (modules.count() > 1)
406  renderedText.append("\t\t</tr>\n");
407 
408  // qDebug("CTextRendering: %s", renderedText.latin1());
409  return renderedText;
410 }
