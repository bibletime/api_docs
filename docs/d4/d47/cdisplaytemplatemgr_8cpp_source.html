50  m_multiModuleHeaders(true) {
51  BT_ASSERT(!m_instance);
52  m_instance = this;

56  const QDir::Filters readableFileFilter(QDir::Files | QDir::Readable);
57  const QDir & td = DU::getDisplayTemplatesDir(); // Global template directory
58  const QDir & utd = DU::getUserDisplayTemplatesDir(); // User template directory
59 
60  // Load regular templates:
61  {
62  const QStringList filter("*.tmpl");
63  // Preload global display templates from disk:
64  for (auto const & file : td.entryList(filter, readableFileFilter))
65  loadTemplate(td.canonicalPath() + "/" + file);
66  // Preload user display templates from disk:
67  for (auto const & file : utd.entryList(filter, readableFileFilter))
68  loadTemplate(utd.canonicalPath() + "/" + file);
69  }
70 
71  if (!m_templateMap.contains(CSSTEMPLATEBASE)) {
72  errorMessage = QObject::tr("CSS base template not found!");
73  return;
74  }
75 
76  // Load CSS templates:
77  {
78  const QStringList cssfilter("*.css");
79  // Load global app stylesheets
80  for (auto const & file : td.entryList(cssfilter, readableFileFilter))
81  loadCSSTemplate(td.canonicalPath() + "/" + file);
82  // Load user app stylesheets
83  for (auto const & file : utd.entryList(cssfilter, readableFileFilter))
84  loadCSSTemplate(utd.canonicalPath() + "/" + file);
85  }
86  }
87 
88  if (!m_cssMap.contains(defaultTemplateName())) {
89  errorMessage = QObject::tr("Default template \"%1\" not found!")
90  .arg(defaultTemplateName());
91  return;
92  }
93 
94  // Create template names cache:
95  m_availableTemplateNamesCache = m_templateMap.keys();
96  const bool b = m_availableTemplateNamesCache.removeOne(CSSTEMPLATEBASE);
97  BT_ASSERT(b);
98  m_availableTemplateNamesCache.append(m_cssMap.keys());
99  std::sort(m_availableTemplateNamesCache.begin(),
100  m_availableTemplateNamesCache.end());
101

105 QString CDisplayTemplateMgr::fillTemplate(const QString & name,
106  const QString & content,
107  const Settings & settings) const
108 {
109  BT_ASSERT(name != CSSTEMPLATEBASE);
110  BT_ASSERT(name.endsWith(".css") || name.endsWith(".tmpl"));
111  BT_ASSERT(!name.endsWith(".css") || m_cssMap.contains(name));
112  BT_ASSERT(!name.endsWith(".tmpl") || m_templateMap.contains(name));
113  const bool templateIsCss = name.endsWith(".css");
114 
115  QString displayTypeString;
116  QString moduleName;
117 
118  if (!settings.pageCSS_ID.isEmpty()) {
119  displayTypeString = settings.pageCSS_ID;
120  moduleName = "";
121  }
122  else {
123  if (settings.modules.count()) {
124  switch (settings.modules.first()->type()) {
125 
126  case CSwordModuleInfo::Bible:
127  displayTypeString = "bible";
128  break;
129 
130  case CSwordModuleInfo::GenericBook:
131  displayTypeString = "book";
132  break;
133 
134  case CSwordModuleInfo::Commentary:
135  case CSwordModuleInfo::Lexicon:
136  default:
137  displayTypeString = "singleentry";
138  break;
139  }
140  moduleName = settings.modules.first()->name();
141  }
142  else { //use bible as default type if no modules are set
143  displayTypeString = "bible";
144  moduleName = "";
145  }
146  }
147 
148  QString newContent = content;
149  const int moduleCount = settings.modules.count();
150 
151  if (moduleCount >= 2 && m_multiModuleHeaders) {
152  //create header for the modules
153  // qDebug() << "There were more than 1 module, create headers";
154  QString header;
155 
156  for (auto const * const mi : settings.modules) {
157  header.append("<th style=\"width:")
158  .append(QString::number(static_cast<int>(100.0 / moduleCount)))
159  .append("%;\">")
160  .append(mi->name())
161  .append("</th>");
162  }
163 
164  newContent = QString("<table><tr>")
165  .append(header)
166  .append("</tr>")
167  .append(content)
168  .append("</table>");
169  }
170 
171  QString langCSS;
172  {
173  const QFont & f = btConfig().getDefaultFont();
174  langCSS.append("#content{font-family:").append(f.family())
175  .append(";font-size:").append(QString::number(f.pointSizeF(), 'f'))
176  .append("pt;font-weight:").append(f.bold() ? "bold" : "normal")
177  .append(";font-style:").append(f.italic() ? "italic" : "normal")
178  .append('}');
179  }
180  {
181  auto const availableLanguages =
182  CSwordBackend::instance()->availableLanguages();
183  BT_ASSERT(availableLanguages);
184  for (auto const & lang : *availableLanguages) {
185  if (lang->abbrev().isEmpty())
186  continue;
187 
188  BtConfig::FontSettingsPair fp = btConfig().getFontForLanguage(*lang);
189  if (fp.first) {
190  const QFont & f = fp.second;
191 
192  /* QFont::weight() returns an int in the range [0, 99] but CSS
193  requires a real number in the range [1, 1000]. No extra
194  checks are needed for floating point precision in the result
195  range. */
196  auto const fontWeight = 1.0 + (f.weight() * 999.0) / 99.0;
197 
198  auto const fontStyleString =
199  [&f]() {
200  switch ((int) f.style()) {
201  case QFont::StyleItalic: return "italic";
202  case QFont::StyleOblique: return "oblique";
203  case QFont::StyleNormal:
204  default:
205  return "normal";
206  }
207  }();
208 
209  auto const textDecorationString =
210  [&f]() {
211  if (f.underline())
212  return f.strikeOut()
213  ? "underline line-through"
214  : "underline";
215  return f.strikeOut() ? "line-through" : "none";
216  }();
217 
218  /// \todo Add support translating more QFont properties to CSS.
219 
220  langCSS.append("*[lang=").append(lang->abbrev()).append("]{")
221  .append("font-family:").append(f.family())
222  .append(";font-size:").append(QString::number(f.pointSizeF(), 'f'))
223  .append("pt;font-weight:").append(QString::number(fontWeight))
224  .append(";font-style:").append(fontStyleString)
225  .append(";text-decoration:").append(textDecorationString)
226  .append('}');
227  }
228  }
229  }
230 
231  namespace DU = util::directory;
232  QString output(m_templateMap[templateIsCss
233  ? QString(CSSTEMPLATEBASE)
234  : name]); // don't change the map's content directly, use a copy
235  output.replace("#TITLE#", settings.title)
236  .replace("#LANG_ABBREV#", settings.langAbbrev)
237  .replace("#DISPLAYTYPE#", displayTypeString)
238  .replace("#LANG_CSS#", langCSS)
239  .replace("#PAGE_DIRECTION#", settings.textDirectionAsHtmlDirAttr())
240  .replace("#CONTENT#", newContent)
241  .replace("#MODTYPE#", displayTypeString)
242  .replace("#MODNAME#", moduleName)
243  .replace("#DISPLAY_TEMPLATES_PATH#", DU::getDisplayTemplatesDir().absolutePath());
244 
245  if (templateIsCss)
246  output.replace("#THEME_STYLE#", readFileToString(m_cssMap[name]));
247 
248  return output;
249 }
