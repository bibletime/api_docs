108  | Qt::ItemIsTristate);
109  m_modsWithoutIndices->setExpanded(true);
110 
111  auto const & modules = CSwordBackend::instance()->moduleList();
112  for (auto const & modulePtr : modules) {
113  if (modulePtr->hasIndex()) {
114  auto item = new QTreeWidgetItem(m_modsWithIndices);
115  item->setText(0, modulePtr->name());
116  item->setText(1, tr("%1 KiB").arg(modulePtr->indexSize() / 1024));
117  item->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled);
118  item->setCheckState(0, Qt::Unchecked);
119  } else {
120  auto item = new QTreeWidgetItem(m_modsWithoutIndices);
121  item->setText(0, modulePtr->name());
122  item->setText(1, tr("0 KiB"));
123  item->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled);
124  item->setCheckState(0, Qt::Checked);
125  }
126  }

157 /** Creates indices for selected modules if no index currently exists */
158 void BtIndexDialog::createIndices() {
159  QList<CSwordModuleInfo *> moduleList;
160 
161  auto & backend = *CSwordBackend::instance();
162  for (int i = 0; i < m_modsWithoutIndices->childCount(); ++i) {
163  if (m_modsWithoutIndices->child(i)->checkState(0) == Qt::Checked) {
164  if (auto * module = backend.findModuleByName(
165  m_modsWithoutIndices->child(i)->text(0).toUtf8()))
166  moduleList.append(module);
167  }
168  }
169 
170  if (!moduleList.isEmpty()) { // This also shows the progress dialog
171  BtModuleIndexDialog::indexAllModules(moduleList);
172  populateModuleList();
173  }
174 }

176 /** Deletes indices for selected modules */
177 void BtIndexDialog::deleteIndices() {
178  bool indicesDeleted = false;
179 
180  auto & backend = *CSwordBackend::instance();
181  for (int i = 0; i < m_modsWithIndices->childCount(); ++i) {
182  if (m_modsWithIndices->child(i)->checkState(0) == Qt::Checked) {
183  if (auto * module = backend.findModuleByName(
184  m_modsWithIndices->child(i)->text(0).toUtf8()))
185  {
186  module->deleteIndex();
187  indicesDeleted = true;
188  }
189  }
190  }
191 
192  // repopulate the list if an action was taken
193  if (indicesDeleted)
194  populateModuleList();
195 }
