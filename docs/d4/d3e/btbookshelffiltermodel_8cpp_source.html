146 bool BtBookshelfFilterModel::nameFilterAcceptsRow(
147  int row,
148  QModelIndex const & parent) const
149 {
150  if (m_nameFilter.isEmpty())
151  return true;
152 
153  auto const * const m = sourceModel();
154  BT_ASSERT(m);
155 
156  auto const itemIndex = m->index(row, m_nameFilterColumn, parent);
157  auto const numChildren = m->rowCount(itemIndex);
158  if (numChildren == 0) {
159  auto const data = m->data(itemIndex, m_nameFilterRole);
160  return data.toString().contains(m_nameFilter, m_nameFilterCase);
161  }
162 
163  for (int i = 0; i < numChildren; ++i)
164  if (filterAcceptsRow(i, itemIndex))
165  return true;
166  return false;
167 }

169 bool BtBookshelfFilterModel::hiddenFilterAcceptsRow(
170  int row,
171  QModelIndex const & parent) const
172 {
173  if (m_showHidden && m_showShown)
174  return true;
175 
176  auto const * const m = sourceModel();
177  BT_ASSERT(m);
178 
179  auto const itemIndex = m->index(row, m_hiddenFilterColumn, parent);
180  auto const numChildren = m->rowCount(itemIndex);
181  if (numChildren == 0) {
182  if (static_cast<Qt::CheckState>(m->data(itemIndex,
183  m_hiddenFilterRole).toBool()))
184  return m_showHidden;
185  return m_showShown;
186  }
187 
188  for (int i = 0; i < numChildren; ++i)
189  if (filterAcceptsRow(i, itemIndex))
190  return true;
191  return false;
192 }

194 bool BtBookshelfFilterModel::categoryFilterAcceptsRow(
195  int row,
196  QModelIndex const & parent) const
197 {
198  if (m_categoryFilter == CSwordModuleInfo::AllCategories)
199  return true;
200
