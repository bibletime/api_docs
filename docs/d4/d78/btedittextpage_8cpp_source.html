99 void BtEditTextPage::initializePage() {
100  m_htmlMode = m_wizard->htmlMode();
101  if (m_htmlMode) {
102  initHtmlFont();
103  m_htmlTextEdit->setHtml(m_text);
104  m_plainTextEdit->clear();
105  }
106  else {
107  m_plainTextEdit->setFont(m_plainTextFont);
108  m_plainTextEdit->setText(m_text);
109 
110  { // Needed to strip html when switching to plain text.
111  QString tmp = m_plainTextEdit->toPlainText();
112  m_plainTextEdit->setText(tmp);
113  }
114 
115  m_htmlTextEdit->clear();
116  }
117  m_plainTextEdit->setAcceptRichText(false);
118  m_htmlTextEdit->setAcceptRichText(true);
119 
120  m_plainTextEdit->setVisible(!m_htmlMode);
121  m_htmlTextEdit->setVisible(m_htmlMode);
122  m_toolBar->setVisible(m_htmlMode);
123 }

261 /** The text's alignment changed. Enable the right buttons. */
262 void BtEditTextPage::alignmentChanged( int a ) {
263  BT_ASSERT(!m_handingFormatChangeFromEditor);
264  bool alignLeft = false;
265  bool alignCenter = false;
266  bool alignRight = false;
267 
268  if (a & Qt::AlignLeft) {
269  alignLeft = true;
270  }
271  else if ((a & Qt::AlignHCenter) || (a & Qt::AlignCenter)) {
272  alignCenter = true;
273  }
274  else if (a & Qt::AlignRight) {
275  alignRight = true;
276  }
277  else {
278  alignLeft = true;
279  qWarning("unknown alignment %i", a);
280  }
281 
282  m_actions.alignLeft->setChecked( alignLeft );
283  m_actions.alignCenter->setChecked( alignCenter );
284  m_actions.alignRight->setChecked( alignRight );
285 }
