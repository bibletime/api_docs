35 void CSwordBibleModuleInfo::initBounds() const {
36  /// \todo The fields calculated by this method could be cached to disk.
37 
38  BT_ASSERT(!m_boundsInitialized);
39 
40  auto & m = module();
41  const bool oldStatus = m.isSkipConsecutiveLinks();
42  m.setSkipConsecutiveLinks(true);
43 
44  m.setPosition(sword::TOP); // position to first entry
45  sword::VerseKey key(m.getKeyText());
46  m_hasOT = (key.getTestament() == 1);
47 
48  m.setPosition(sword::BOTTOM);
49  key = m.getKeyText();
50  m_hasNT = (key.getTestament() == 2);
51 
52  m.setSkipConsecutiveLinks(oldStatus);

56 
57  m_boundsInitialized = true;
58 }

62 QStringList const & CSwordBibleModuleInfo::books() const {
63  {
64  CSwordBackend & b = backend();
65  if (m_cachedLocale != b.booknameLanguage()) {
66  // Reset the booklist because the locale has changed
67  m_cachedLocale = b.booknameLanguage();
68  m_bookList.reset();
69  }
70  }
71 
72  if (!m_bookList) {
73  m_bookList.emplace();
74 
75  // Initialize m_hasOT and m_hasNT
76  if (!m_boundsInitialized)
77  initBounds();
78 
79  int min = 1; // 1 = OT
80  int max = 2; // 2 = NT
81 
82  if (!m_hasOT)
83  min++; // min == 2
84 
85  if (!m_hasNT)
86  max--; // max == 1
87 
88  if (min > max) {
89  qWarning("CSwordBibleModuleInfo (%s) no OT and not NT! Check your config!", module().getName());
90  } else {
91  std::unique_ptr<sword::VerseKey> key(
92  static_cast<sword::VerseKey *>(module().createKey()));
93  key->setPosition(sword::TOP);
94 
95  for (key->setTestament(min); !key->popError() && key->getTestament() <= max; key->setBook(key->getBook() + 1)) {
96  m_bookList->append( QString::fromUtf8(key->getBookName()) );
97  }
98  }
99  }
100 
101  return *m_bookList;
102 }
