99bool CSwordLDKey::setKey(const char *newKey) {
100 BT_ASSERT(newKey);
101
102 if (newKey) {
103 m_key = newKey; //set the key
104 m_module->swordModule().setKey(&m_key);
105 m_module->snap();
106 }
107
108 return !m_key.popError();
