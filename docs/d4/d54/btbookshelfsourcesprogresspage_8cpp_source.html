97void BtBookshelfSourcesProgressPage::initializePage() {
98 destroyThread();
99
100 m_installCompleted = false;
101 m_thread = new BtSourcesThread(this);
102 BT_CONNECT(m_thread, &BtSourcesThread::percentComplete,
103 m_progressBar, &QProgressBar::setValue,
104 Qt::QueuedConnection);
105 BT_CONNECT(m_thread, &BtSourcesThread::showMessage,
106 m_msgLabel, &QLabel::setText,
107 Qt::QueuedConnection);
108 BT_CONNECT(m_thread, &BtSourcesThread::finished,
109 this, &BtBookshelfSourcesProgressPage::slotThreadFinished,
110 Qt::QueuedConnection);
111 m_thread->start();
112 m_stopButton->setEnabled(true);
113 btWizard().downloadStarted();
114 retranslateUi();
115}
