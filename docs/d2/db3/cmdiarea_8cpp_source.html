94 QMdiSubWindow * CMDIArea::addSubWindow(QWidget * widget,
95  Qt::WindowFlags windowFlags)
96 {
97  QMdiSubWindow * const subWindow = QMdiArea::addSubWindow(widget, windowFlags);
98  subWindow->installEventFilter(this);
99  fixSystemMenu(subWindow);
100 
101  // Manual arrangement mode
102  enableWindowMinMaxFlags(true);
103  if (m_mdiArrangementMode == ArrangementModeManual) {
104  // Note that the window size/maximization may be changed later by a session restore.
105  // If we already have an active window, make the new one simular to it
106  if (activeSubWindow()) {
107  if (activeSubWindow()->isMaximized()) {
108  subWindow->showMaximized(); // Maximize the new window
109  } else {
110  // Make new window the same size as the active window and move it slightly.
111  subWindow->resize(activeSubWindow()->size());
112  QRect subWinGeom = activeSubWindow()->geometry();
113  static const int MOVESIZE = 30;
114  subWinGeom.translate(MOVESIZE, MOVESIZE);
115  // If it goes off screen, move it almost to the top left
116  if (!frameRect().contains(subWinGeom))
117  subWinGeom.moveTo(MOVESIZE, MOVESIZE);
118  subWindow->setGeometry(subWinGeom);
119  }
120  } else {
121  //set the window to be big enough
122  subWindow->resize(400, 400);
123  }
124  subWindow->raise();
125  } else {
126  // Automatic arrangement modes
127  enableWindowMinMaxFlags(false);
128  triggerWindowUpdate();
129  }
130  return subWindow;
131 }

133 void CMDIArea::setMDIArrangementMode( const MDIArrangementMode newArrangementMode ) {
134  m_mdiArrangementMode = newArrangementMode;
135  switch (m_mdiArrangementMode) {
136  case ArrangementModeManual:
137  setViewMode(QMdiArea::SubWindowView);

151  break;
152  }

154  QObject* parent = tab->parent();
155  if (parent == this)
156  tab->setTabsClosable(true);
157  }

171  }

174  return;
175  }
176  setViewMode(QMdiArea::SubWindowView);
177 
178  setUpdatesEnabled(false);
179 
180  QMdiSubWindow * const active = activeSubWindow();
181 
182  const int widthForEach = width() / windows.count();
183  int x = 0;
184  for (auto * const window : windows) {
185  window->showNormal();
186 
187  const int preferredWidth = window->minimumWidth() + window->baseSize().width();
188  const int actWidth = qMax(widthForEach, preferredWidth);
189 
190  window->setGeometry(x, 0, actWidth, height());
191  x += actWidth;

236 void CMDIArea::myTile() {
237  if (!updatesEnabled()) {
238  return;
239  }
240  QList<QMdiSubWindow*> windows = usableWindowList();
241  if (windows.isEmpty()) {
242  return;
243  }
244  setViewMode(QMdiArea::SubWindowView);
245 
246  setUpdatesEnabled(false);
247  QMdiSubWindow * const active = activeSubWindow();
248 
249  const QRect domain = contentsRect();
250  const int n = windows.size();
251  const int ncols = qMax(qCeil(qSqrt(qreal(n))), 1);
252  const int nrows = qMax((n % ncols) ? (n / ncols + 1) : (n / ncols), 1);
253  const int nspecial = (n % ncols) ? (ncols - n % ncols) : 0;
254  const int dx = domain.width() / ncols;
255  const int dy = domain.height() / nrows;
256 
257  int i = 0;
258  for (int row = 0; row < nrows; ++row) {
259  const int y1 = int(row * (dy + 1));
260  for (int col = 0; col < ncols; ++col) {
261  if (row == 1 && col < nspecial)
262  continue;
263  const int x1 = int(col * (dx + 1));
264  int x2 = int(x1 + dx);
265  int y2 = int(y1 + dy);
266  if (row == 0 && col < nspecial) {
267  y2 *= 2;
268  if (nrows != 2)
269  y2 += 1;
270  else
271  y2 = domain.bottom();
272  }
273  if (col == ncols - 1 && x2 != domain.right())
274  x2 = domain.right();
275  if (row == nrows - 1 && y2 != domain.bottom())
276  y2 = domain.bottom();
277  QWidget *widget = windows.at(i++);
278  QRect newGeometry = QRect(QPoint(x1, y1), QPoint(x2, y2));
279  widget->setGeometry(QStyle::visualRect(widget->layoutDirection(), domain, newGeometry));
280  }
281  }
282 
283  if (active != nullptr) {
284  active->setFocus();
285  }
286 
287  setUpdatesEnabled(true);
288  emitWindowCaptionChanged();
289 }

291 void CMDIArea::myCascade() {
292  if (!updatesEnabled()) {
293  return;
294  }
295  QList<QMdiSubWindow*> windows = usableWindowList();
296  if (windows.isEmpty()) {
297  return;
298  }
299  setViewMode(QMdiArea::SubWindowView);
300 
301  if (windows.count() == 1) {
302  windows.first()->showMaximized();
303  }
304  else {
305  setUpdatesEnabled(false);
306 
307  QMdiSubWindow * const active = activeSubWindow();
308 
309  static const unsigned offsetX = 40;
310  static const unsigned offsetY = 40;
311  const unsigned int windowWidth = width() - (windows.count() - 1) * offsetX;
312  const unsigned int windowHeight = height() - (windows.count() - 1) * offsetY;
313  unsigned int x = 0;
314  unsigned int y = 0;
315 
316  for (auto * const window : windows) {
317  if (window == active) { //leave out the active window which should be the top window
318  continue;
319  }
320  window->showNormal();
321  window->raise(); //make it the on-top-of-window-stack window to make sure they're in the right order
322  window->setGeometry(x, y, windowWidth, windowHeight);
323  x += offsetX;
324  y += offsetY;
325  }
326  active->showNormal();
327  active->setGeometry(x, y, windowWidth, windowHeight);
328  active->raise();
329  active->activateWindow();
330 
331  setUpdatesEnabled(true);
332  }
333 
334  emitWindowCaptionChanged();
335 }

392 bool CMDIArea::eventFilter(QObject *o, QEvent *e) {
393  const QMdiSubWindow * const w = qobject_cast<QMdiSubWindow*>(o);
394 
395  // Let the event be handled by other filters:
396  if (w == nullptr)
397  return QMdiArea::eventFilter(o, e);
398 
399  switch (e->type()) {
400  case QEvent::WindowStateChange: {
401  Qt::WindowStates const newState(w->windowState());
402  Qt::WindowStates const oldState(
403  static_cast<QWindowStateChangeEvent *>(e)->oldState());
404 
405  /*
406  Do not handle window activation or deactivation here, it will
407  produce wrong results because this event is handled too early. Let
408  lambda connected to subWindowActivated() handle this.
409  */
410 
411  // Check if subwindow was maximized or un-maximized:
412  if ((newState ^ oldState) & Qt::WindowMaximized) {
413  triggerWindowUpdate();
414  break;
415  }
416 
417  // Check if subwindow was minimized or de-minimized:
418  if ((newState ^ oldState) & Qt::WindowMinimized) {
419  triggerWindowUpdate();
420  }
421  break;
422  }
423  case QEvent::Close:
424  triggerWindowUpdate();
425  break;
426  case QEvent::WindowTitleChange:
427  if (o == activeSubWindow()) {
428  Q_EMIT sigSetToplevelCaption(w->windowTitle());
429  }
430  return QMdiArea::eventFilter(o, e);
431  break;
432  default:
433  break;
434  }
435 
436  return false; // Don't filter the event out
437 }
