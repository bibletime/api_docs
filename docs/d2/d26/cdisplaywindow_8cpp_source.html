159 void CDisplayWindow::storeProfileSettings(BtConfigCore & conf) const {
160  QWidget const * const w = getProfileWindow(parentWidget());
161  BT_ASSERT(w);
162 
163  /**
164  \note We don't use saveGeometry/restoreGeometry for MDI subwindows,
165  because they give slightly incorrect results with some window
166  managers. Might be related to Qt bug QTBUG-7634.
167  */
168  const QRect rect(w->x(), w->y(), w->width(), w->height());
169  conf.setValue<QRect>("windowRect", rect);
170  conf.setValue<bool>("staysOnTop",
171  w->windowFlags() & Qt::WindowStaysOnTopHint);
172  conf.setValue<bool>("staysOnBottom",
173  w->windowFlags() & Qt::WindowStaysOnBottomHint);
174  conf.setValue("maximized", w->isMaximized());
175 
176  bool hasFocus = (w == dynamic_cast<CDisplayWindow *>(mdi()->activeSubWindow()));
177  conf.setValue("hasFocus", hasFocus);
178  // conf.setSessionValue("type", static_cast<int>(modules().first()->type()));
179 
180  // Save current key:
181  if (auto const * const k = m_swordKey)
182  conf.setValue("key", k->normalizedKey());
183 
184  // Save list of modules:
185  conf.setValue("modules", m_modules);
186 
187  // Default for "not a write window":
188  conf.setValue("writeWindowType", int(0));

632 void CDisplayWindow::reload(CSwordBackend::SetupChangedReason) {
633  { // First make sure all used Sword modules are still present:
634  CSwordBackend & backend = *(CSwordBackend::instance());
635  QMutableStringListIterator it(m_modules);
636  while (it.hasNext())
637  if (!backend.findModuleByName(it.next()))
638  it.remove();
639  }
640 
641  if (m_modules.isEmpty()) {
642  close();
643  } else {
644  m_displayWidget->reloadModules();
645 
646  if (CKeyChooser * const kc = keyChooser())
647  kc->setModules(modules(), false);
648 
649  lookup();
650 
651  m_actionCollection->readShortcuts("Displaywindow shortcuts");
