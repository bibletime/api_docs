108 void CSearchDialog::startSearch() {
109  QString originalSearchText(m_searchOptionsArea->searchText());
110 
111  // first check the search string for errors
112  {
113  QString TestString(originalSearchText);
114  QRegExp ReservedWords("heading:|footnote:|morph:|strong:");
115  if (TestString.replace(ReservedWords, "").simplified().isEmpty()) {
116  return;
117  }
118  }
119  QString searchText = CSwordModuleSearch::prepareSearchText(originalSearchText, m_searchOptionsArea->searchType());
120 
121  // Insert search text into history list of combobox
122  m_searchOptionsArea->addToHistory(originalSearchText);
123 
124  auto const & searchModules = m_searchOptionsArea->modules();
125 
126  // Check that we have the indices we need for searching
127  /// \warning indexing is some kind of internal optimization, so we leave
128  /// modules const, but unconst them here only
129  QList<CSwordModuleInfo*> unindexedModules;
130  for (auto const * const m : searchModules)
131  if (!m->hasIndex())
132  unindexedModules.append(const_cast<CSwordModuleInfo*>(m));
133 
134  if (unindexedModules.size() > 0) {
135  // Build the list of module names:
136  QStringList moduleNameList;
137  for (auto const * const m : unindexedModules)
138  moduleNameList.append(m->name());
139  QString moduleNames("<br><center>");
140  moduleNames.append(moduleNameList.join(", "));
141  moduleNames.append("</center><br>");
142 
143  // Ask the user about unindexed modules:
144  int result = message::showQuestion(
145  this, tr("Missing indices"),
146  tr("The following modules need to be indexed before they can be"
147  " searched in:") + moduleNames + tr("Indexing could take a l"
148  "ong time. Click \"Yes\" to index the modules and start the "
149  "search, or \"No\" to cancel the search."),
150  QMessageBox::Yes | QMessageBox::No, QMessageBox::Yes);
151 
152  // User didn't press "Yes":
153  if ((result & (QMessageBox::Yes | QMessageBox::Default)) == 0x0) {
154  return;
155  }
156 
157  // Show indexing dialog, and index the modules:
158  if (!BtModuleIndexDialog::indexAllModules(unindexedModules)) {
159  // Failed or user cancelled.
160  return;
161  }
162  }
163 
164  // Disable the dialog:
165  setEnabled(false);
166  setCursor(Qt::WaitCursor);
167 
168  // Execute search:
169  CSwordModuleSearch::Results searchResult;
170  try {
171  searchResult =
172  CSwordModuleSearch::search(searchText,
173  searchModules,
174  m_searchOptionsArea->searchScope());
175  } catch (...) {
176  QString msg;
177  try {
178  throw;
179  } catch (std::exception const & e) {
180  msg = e.what();
181  } catch (...) {
182  msg = tr("<UNKNOWN EXCEPTION>");
183  }
184 
185  message::showWarning(this,
186  tr("Search aborted"),
187  tr("An internal error occurred while executing "
188  "your search:<br/><br/>%1").arg(msg));
189  // Re-enable the dialog:
190  setEnabled(true);
191  setCursor(Qt::ArrowCursor);
192  return;
193  }
194 
195  // Display the search results:
196  if (!searchResult.empty()) {
197  m_searchResultArea->setSearchResult(m_searchOptionsArea->searchText(),
198  std::move(searchResult));
199  m_analyseButton->setEnabled(true);
200  } else {
201  m_searchResultArea->reset();
202  }
203  raise();
204  activateWindow();
205 
206  // Re-enable the dialog:
207  setEnabled(true);
208  setCursor(Qt::ArrowCursor);
209 }

211 void CSearchDialog::reset(BtConstModuleList modules, QString const & searchText)
212 {
213  m_searchOptionsArea->reset();
214  m_searchResultArea->reset();
215  m_analyseButton->setEnabled(false);
216 
217  auto const haveModules = !modules.isEmpty();
218  if (haveModules) {
219  m_searchOptionsArea->setModules(std::move(modules));
220  } else {
221  m_searchOptionsArea->chooseModules();
222  }
223 
224  m_searchOptionsArea->setSearchText(searchText);
225  if (isHidden())
226  show();
227 
228  if (haveModules && !searchText.isEmpty())
229  startSearch();
230 
231  // moved these to after the startSearch() because
232  // the progress dialog caused them to loose focus.
233  raise();
234  activateWindow();
235 }
