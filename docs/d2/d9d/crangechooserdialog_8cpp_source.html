128 void CRangeChooserDialog::initConnections() {
129  BT_CONNECT(m_rangeList, &QListWidget::currentItemChanged,
130  [this](QListWidgetItem *, QListWidgetItem * const previous) {
131  if (previous) {
132  BT_ASSERT(dynamic_cast<RangeItem *>(previous));
133  saveCurrentToRange(static_cast<RangeItem*>(previous));
134  }
135  resetEditControls();
136  });
137  BT_CONNECT(m_nameEdit, &QLineEdit::textEdited,
138  this, &CRangeChooserDialog::nameEditTextChanged);
139  BT_CONNECT(m_rangeEdit, &QTextEdit::textChanged,
140  [this]{
141  m_resultList->clear();
142  auto const range =
143  m_rangeEdit->toPlainText().replace(
144  QRegExp("\\s{0,}-\\s{0,}"),
145  "-");
146 
147  auto const & backend = CSwordBackend::instance();
148  for (auto const & moduleName : m_scopeModules) {
149  auto * const module =
150  backend->findModuleByName(moduleName);
151  if (!module)
152  continue;
153  auto const verses(
154  sword::VerseKey(module->module().getKey())
155  .parseVerseList(
156  range.toUtf8().constData(),
157  "Genesis 1:1",
158  true));
159  if (verses.getCount() > 0) {
160  for (int i = 0; i < verses.getCount(); i++) {
161  auto const * const elementText =
162  verses.getElement(i)->getRangeText();
163  new QListWidgetItem(
164  QString::fromUtf8(elementText),
165  m_resultList);
166  }
167  break;
168  }
169  }
170  });
171 
172  // Connect buttons:
173  BT_CONNECT(m_buttonBox, &QDialogButtonBox::accepted,
174  this, &CRangeChooserDialog::accept);
175  BT_CONNECT(m_buttonBox, &QDialogButtonBox::rejected,
176  this, &CRangeChooserDialog::reject);
177  BT_CONNECT(m_newRangeButton, &QPushButton::clicked,
178  [this]{
179  m_rangeList->setCurrentItem(
180  new RangeItem(tr("New range"),
181  QString(),
182  m_rangeList));
183  resetEditControls();
184  });
185  BT_CONNECT(m_deleteRangeButton, &QPushButton::clicked,
186  [this]{
187  BT_ASSERT(dynamic_cast<RangeItem *>(
188  m_rangeList->currentItem()));
189  QListWidgetItem * const i = m_rangeList->currentItem();
190  m_rangeList->removeItemWidget(i);
191  delete i;
192 
193  resetEditControls();
194  });
195  QPushButton * defaultsButton = m_buttonBox->button(QDialogButtonBox::RestoreDefaults);
196  BT_CONNECT(defaultsButton, &QPushButton::clicked,
197  [this]{
198  m_rangeList->clear();
199  btConfig().deleteSearchScopesWithCurrentLocale();
200  auto const map(
201  btConfig().getSearchScopesForCurrentLocale(
202  m_scopeModules));
203  for (auto it = map.begin(); it != map.end(); ++it)
204  new RangeItem(it.key(), it.value(), m_rangeList);
205  m_rangeList->setCurrentItem(nullptr);
206  resetEditControls();
207  });
208 }

246 void CRangeChooserDialog::resetEditControls() {
247  const QListWidgetItem * const item = m_rangeList->currentItem();
248  BT_ASSERT(!item || dynamic_cast<RangeItem const *>(item));
249  const RangeItem * rangeItem = static_cast<const RangeItem *>(item);
250 
251  m_nameEdit->setEnabled(item != nullptr);
252  m_rangeEdit->setEnabled(item != nullptr);
253  m_resultList->setEnabled(item != nullptr);
254  m_deleteRangeButton->setEnabled(item != nullptr);
255  m_nameEdit->setText(item != nullptr ? rangeItem->caption() : "");
256  m_rangeEdit->setText(item != nullptr ? rangeItem->range() : "");
257 
258  if (item != nullptr)
259  m_nameEdit->setFocus();
260 
261  nameEditTextChanged(item != nullptr ? rangeItem->caption() : "");
262 }

264 void CRangeChooserDialog::accept() {
265  // Update the active item:
266  QListWidgetItem *currentItem = m_rangeList->currentItem();
267  if (currentItem != nullptr) {
268  BT_ASSERT(dynamic_cast<RangeItem *>(currentItem));
269  saveCurrentToRange(static_cast<RangeItem*>(currentItem));
270  }
271 
272  // Save the new sorted map of search scopes:
273  m_rangeList->sortItems();
274  BtConfig::StringMap map;
275  for (int i = 0; i < m_rangeList->count(); i++) {
276  BT_ASSERT(dynamic_cast<RangeItem *>(m_rangeList->item(i)));
277  const RangeItem * item = static_cast<RangeItem*>(m_rangeList->item(i));
278  map[item->caption()] = item->range();
279  }
280  btConfig().setSearchScopesWithCurrentLocale(m_scopeModules, map);
281 
282  QDialog::accept();
283 }
