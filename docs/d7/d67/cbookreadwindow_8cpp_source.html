86 void CBookReadWindow::initView() {

92

95 
96  // Create Navigation toolbar
97  setMainToolBar( new QToolBar(this) );
98  mainToolBar()->setAllowedAreas(Qt::TopToolBarArea);
99  mainToolBar()->setFloatable(false);
100  addToolBar(mainToolBar());
101  setKeyChooser( CKeyChooser::createInstance(modules(), history(), key(), mainToolBar()) );
102 
103  // Create the Works toolbar
104  setModuleChooserBar( new BtModuleChooserBar(this));
105  moduleChooserBar()->setModules(getModuleList(), modules().first()->type(), this);
106  addToolBar(moduleChooserBar());
107 
108  // Create the Tools toolbar
109  setButtonsToolBar( new QToolBar(this) );
110  buttonsToolBar()->setAllowedAreas(Qt::TopToolBarArea);
111  buttonsToolBar()->setFloatable(false);
112  addToolBar(buttonsToolBar());
113 
114  setCentralWidget( splitter );
115  setWindowIcon(util::tool::getIconForModule(modules().first()));
