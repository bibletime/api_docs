44CBookReadWindow::CBookReadWindow(QList<CSwordModuleInfo *> const & modules,
45 QString const & key,
46 CMDIArea * parent)
47 : CDisplayWindow(modules, key, false, new ActionCollection(), parent)
48{
49 auto * const h = history();
50
51 m_treeChooserDock = new QDockWidget(this);
52 m_treeChooserDock->setFeatures(QDockWidget::NoDockWidgetFeatures);
53 m_treeChooserDock->setAllowedAreas(
54 Qt::LeftDockWidgetArea | Qt::RightDockWidgetArea);
55 // "Remove" title bar:
56 m_treeChooserDock->setTitleBarWidget(new QWidget(m_treeChooserDock));
57 m_treeChooser =
58 new CBookTreeChooser(constModules(),
59 swordKey(),
60 m_treeChooserDock);
61 BT_CONNECT(m_treeChooser, &CKeyChooser::keyChanged,
62 h, &BTHistory::add);
63 BT_CONNECT(h, &BTHistory::historyMoved,
64 m_treeChooser, &CKeyChooser::handleHistoryMoved);
65 m_treeChooserDock->setWidget(m_treeChooser);
66 m_treeChooserDock->hide();
67 addDockWidget(Qt::LeftDockWidgetArea, m_treeChooserDock);
68
69 init();
70}

130void CBookReadWindow::setupMainWindowToolBars() {

132
133 // Navigation toolbar

142 h, &BTHistory::add);
143 BT_CONNECT(h, &BTHistory::historyMoved,
144 keyChooser, &CKeyChooser::handleHistoryMoved);
145 btMainWindow()->navToolBar()->addWidget(keyChooser);
146 BT_CONNECT(keyChooser, &CKeyChooser::keyChanged,
147 this, &CBookReadWindow::lookupSwordKey);
148 BT_CONNECT(this, &CBookReadWindow::sigKeyChanged,
149 keyChooser, &CKeyChooser::updateKey);
150
151 // Works toolbar
152 btMainWindow()->worksToolBar()->associateWithWindow(this);
153
154 // Tools toolbar
155 btMainWindow()->toolsToolBar()->addAction(m_treeAction); // Tree
156 BtDisplaySettingsButton* button = new BtDisplaySettingsButton(buttonsToolBar());
