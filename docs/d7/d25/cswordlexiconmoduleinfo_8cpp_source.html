156 QString CSwordLexiconModuleInfo::normalizeStrongsKey(const QString &key) const {
157 
158  QRegExp rx("([GH]*)([0-9]+)");
159  if (! rx.exactMatch(key))
160  return key;
161  QString StrongsChar = rx.cap(1);
162  QString digits = rx.cap(2);
163 
164  while (digits.length() < m_strongsDigitsLength)
165  digits = "0" +digits;
166 
167  while (digits.length() > m_strongsDigitsLength && digits.at(0) == "0")
168  digits = digits.right(digits.length()-1);
169 
170  QString newKey = digits;
171  if (m_hasLeadingStrongsLetter)
172  newKey = StrongsChar + digits;
173  return newKey;
174 }
