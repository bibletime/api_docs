40 QIcon const & BtModuleChooserButton::icon() {
41  switch (m_popup->moduleType()) {
42  case CSwordModuleInfo::Bible:
43  return m_popup->selectedModule().isEmpty()
44  ? CResMgr::modules::bible::icon_add()
45  : CResMgr::modules::bible::icon_unlocked();
46  case CSwordModuleInfo::Commentary:
47  return m_popup->selectedModule().isEmpty()
48  ? CResMgr::modules::commentary::icon_add()
49  : CResMgr::modules::commentary::icon_unlocked();
50  case CSwordModuleInfo::Lexicon:
51  return m_popup->selectedModule().isEmpty()

53  : CResMgr::modules::lexicon::icon_unlocked();

57  : CResMgr::modules::book::icon_unlocked();

86 /** Is called after a module was selected in the popup */
87 void BtModuleChooserButton::moduleChosen(
88  CSwordModuleInfo const * const newModule)
89 {
90  // If no module was previously selected:
91  auto const & selectedModule = m_popup->selectedModule();
92  if (selectedModule.isEmpty()) {
93  if (newModule) {
94  m_popup->setSelectedModule(newModule->name());
95  setIcon(newModule->moduleIcon());
96  Q_EMIT sigModuleAdd(m_popup->buttonIndex(), newModule->name());
97  }
98  } else {
99  if (newModule) {
100  m_popup->setSelectedModule(newModule->name());
101  setIcon(newModule->moduleIcon());
102  Q_EMIT sigModuleReplace(m_popup->buttonIndex(), newModule->name());
103  } else {
104  m_popup->setSelectedModule(QString());
105  setIcon(icon());
106  Q_EMIT sigModuleRemove(m_popup->buttonIndex());
107  }
108  }
