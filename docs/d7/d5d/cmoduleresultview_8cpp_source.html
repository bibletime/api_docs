145 void CModuleResultView::setupTree(const CSwordModuleSearch::Results & results,
146  const QString & searchedText)
147 {
148  /// \todo implement sorting in this method.
149 
150  clear();
151  m_results.clear();
152  qDeleteAll(m_strongsResults);
153  m_strongsResults.clear();
154 
155  bool strongsAvailable = false;
156 
157  for (auto const & result : results) {
158  auto const * const m = result.module;
159  BT_ASSERT(!m_results.contains(m));
160  m_results.insert(m, result.results);
161  QTreeWidgetItem * const item =
162  new QTreeWidgetItem(this,
163  QStringList(m->name())
164  << QString::number(result.results.size()));
165 
166  item->setIcon(0, util::tool::getIconForModule(m));
167  /*
168  We need to make a decision here. Either don't show any Strong's
169  number translations, or show the first one in the search text, or
170  figure out how to show them all. I choose option number 2 at this time.
171  */
172 
173  // strong search text index for finding "strong:"
174  int sstIndex = searchedText.indexOf("strong:", 0);
175  if (sstIndex != -1) {
176  /*
177  Get the strongs number from the search text. First find the first
178  space after "strong:". This should indicate a change in search
179  token
180  */
181  sstIndex += 7;
182  const int sTokenIndex = searchedText.indexOf(" ", sstIndex);
183  const QString sNumber(searchedText.mid(sstIndex, sTokenIndex - sstIndex));
184 
185  setupStrongsResults(m, result.results, item, sNumber);
186 
187  /// \todo item->setOpen(true);
188  strongsAvailable = true;
189  }
190  };
191 
192  // Allow to hide the module strongs if there are any available
193  setRootIsDecorated( strongsAvailable );
194 }

214 void CModuleResultView::executed( QTreeWidgetItem* i, QTreeWidgetItem*) {
215  QString itemText;
216 
217  if (!i) {
218  //Clear list
219  Q_EMIT moduleChanged();
220  return;
221  }
222  if (CSwordModuleInfo *m = CSwordBackend::instance()->findModuleByName(i->text(0))) {
223  Q_EMIT moduleChanged();
224  Q_EMIT moduleSelected(m, m_results.value(m));
225  return;
226  }
227 
228  StrongsResultList *strongsResult = m_strongsResults[activeModule()];
229 
230  if (!strongsResult) {
231  return;
232  }
233 
234  itemText = i->text(0);
235  for (int cnt = 0; cnt < strongsResult->count(); cnt++) {
236  if (strongsResult->at(cnt).keyText() == itemText) {
237  //clear the verses list
238  Q_EMIT moduleChanged();
239  Q_EMIT strongsSelected(activeModule(),
240  strongsResult->at(cnt).getKeyList());
241  return;
242  }
243  }
244 }
