289  /** Loads a list of items (with subitem trees) from a named file
290  * or from the default bookmarks file. */
291  QList<BookmarkItemBase *> loadTree(QString fileName = QString()) {
292  QList<BookmarkItemBase*> itemList;
293 
294  QDomDocument doc;
295  QString bookmarksFile = loadXmlFromFile(fileName);
296  if (bookmarksFile.isNull())
297  return QList<BookmarkItemBase*>();
298 
299  doc.setContent(bookmarksFile);
300  QDomElement document = doc.documentElement();
301  if ( document.tagName() != "SwordBookmarks" ) {
302  qWarning("Not a BibleTime Bookmark XML file");
303  return QList<BookmarkItemBase*>();
304  }
305 
306  QDomElement child = document.firstChild().toElement();
307 
308  while ( !child.isNull() && child.parentNode() == document) {
309  BookmarkItemBase* i = handleXmlElement(child, nullptr);
310  itemList.append(i);
311  if (!child.nextSibling().isNull()) {
312  child = child.nextSibling().toElement();
313  }
314  else {
315  child = QDomElement(); //null
316  }
317 
318  }
319 
320  return itemList;
321  }
322 
323  /** Create a new item from a document element. */
324  BookmarkItemBase * handleXmlElement(QDomElement & element, BookmarkItemBase * parent) {
325  BookmarkItemBase* newItem = nullptr;
326  if (element.tagName() == "Folder") {
327  BookmarkFolder* newFolder = new BookmarkFolder(QString(), parent);
328  if (element.hasAttribute("caption")) {
329  newFolder->setText(element.attribute("caption"));
330  }
331  QDomNodeList childList = element.childNodes();
332  for (int i = 0; i < childList.length(); i++) {
333  QDomElement newElement = childList.at(i).toElement();
334  handleXmlElement(newElement, newFolder); // passing parent in constructor will add items to tree
335  }
336  newItem = newFolder;
337  }
338  else if (element.tagName() == "Bookmark") {
339  BookmarkItem* newBookmarkItem = new BookmarkItem(parent);
340  if (element.hasAttribute("modulename")) {
341  //we use the name in all cases, even if the module isn't installed anymore
342  newBookmarkItem->setModule(element.attribute("modulename"));
343  }
344  if (element.hasAttribute("key")) {
345  newBookmarkItem->setKey(element.attribute("key"));
346  }
347  if (element.hasAttribute("description")) {
348  newBookmarkItem->setDescription(element.attribute("description"));
349  }
350  if (element.hasAttribute("title")) {
351  newBookmarkItem->setText(element.attribute("title"));
352  }
353  newItem = newBookmarkItem;
354  }
355  return newItem;
356  }

358  /** Loads a bookmark XML document from a named file or from the default bookmarks file. */
359  QString loadXmlFromFile(QString fileName = QString()) {
360 
361  if (fileName.isEmpty())
362  fileName = defaultBookmarksFile();
363 
364  QFile file(fileName);
365  if (!file.exists())
366  return QString();
367 
368  QString xml;
369  if (file.open(QIODevice::ReadOnly)) {
370  QTextStream t;
371  t.setAutoDetectUnicode(false);
372  t.setCodec(QTextCodec::codecForName("UTF-8"));
373  t.setDevice(&file);
374  xml = t.readAll();
375  file.close();
376  }
377  return xml;
378  }

400  /** Writes one item to a document element. */
401  void saveItem(BookmarkItemBase * item, QDomElement & parentElement) {
402  BookmarkFolder* folderItem = nullptr;
403  BookmarkItem* bookmarkItem = nullptr;
404 
405  if ((folderItem = dynamic_cast<BookmarkFolder*>(item))) {
406  QDomElement elem = parentElement.ownerDocument().createElement("Folder");
407  elem.setAttribute("caption", folderItem->text());
408 
409  parentElement.appendChild(elem);
410 
411  for (int i = 0; i < folderItem->childCount(); i++) {
412  saveItem(folderItem->child(i), elem);
413  }
414  }
415  else if ((bookmarkItem = dynamic_cast<BookmarkItem*>(item))) {
416  QDomElement elem = parentElement.ownerDocument().createElement("Bookmark");
417 
418  elem.setAttribute("key", bookmarkItem->englishKey());
419  elem.setAttribute("description", bookmarkItem->description());
420  elem.setAttribute("modulename", bookmarkItem->moduleName());
421  elem.setAttribute("moduledescription", bookmarkItem->module() ? bookmarkItem->module()->config(CSwordModuleInfo::Description) : QString());
422  if (!bookmarkItem->text().isEmpty()) {
423  elem.setAttribute("title", bookmarkItem->text());
424  }
425  parentElement.appendChild(elem);
426  }
427  }

802  d->itemAs<BookmarkFolder const>(index))

821  if (!bookmarksOnly && toCopy.count() > 1) {
822  // wrong amount of items
823  moreThanOneFolder = true;
824  }

827  return QModelIndexList();

949  Qt::SortOrder const order)
950 {
951  Q_D(BtBookmarksModel);
952

954  QList<BookmarkFolder *> parents;

956  QList<BookmarkItemBase *> items;

959  items.append(items[i]->children());
960  if(BookmarkFolder * ff = dynamic_cast<BookmarkFolder *>(items[i]))
961  parents.append(ff);

965  parents.append(f);

968  Q_EMIT layoutAboutToBeChanged();
969 
970  QModelIndexList indexes;
971  for(int i = 0; i < f->children().size(); ++i)
972  indexes.append(createIndex(i, 0, f->children()[i]));
973 
974  std::sort(f->children().begin(),
975  f->children().end(),
976  order == Qt::AscendingOrder
977  ? BtBookmarksModelSortAscending
978  : BtBookmarksModelSortDescending);
979 
980  for(int i = 0; i < f->children().size(); ++i) {
981  BookmarkItemBase * iii = f->children()[i];
982  for(int ii = 0; ii < indexes.size(); ++ii)
983  if(iii == indexes[ii].internalPointer())
984  changePersistentIndex(createIndex(ii, 0, iii), createIndex(i, 0, iii));
