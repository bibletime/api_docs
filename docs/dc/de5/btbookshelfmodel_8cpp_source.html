40 QVariant BtBookshelfModel::data(CSwordModuleInfo * module, int role) const {
41  BT_ASSERT(module);
42  switch (role) {
43  case Qt::DisplayRole:
44  case ModuleNameRole:
45  return module->name();
46  case Qt::DecorationRole:
47  case ModuleIconRole:
48  return CSwordModuleInfo::moduleIcon(*module);
49  case ModulePointerRole:
50  return QVariant::fromValue(static_cast<void *>(module));
51  case ModuleCategoryRole:
52  return QVariant::fromValue(module->category());
53  case ModuleLanguageRole:
54  return QVariant(); /// \todo Unimplemented
55  case ModuleHiddenRole:
56  return module->isHidden();
57  case ModuleInstallPathRole:
58  return module->config(CSwordModuleInfo::AbsoluteDataPath);
59  case ModuleHasIndexRole:
60  return module->hasIndex();
61  case ModuleIndexSizeRole:
62  return static_cast<qulonglong>(module->indexSize());
63  case ModuleDescriptionRole:
64  return module->config(CSwordModuleInfo::Description);
65  case Qt::ToolTipRole:
66  return QString("<b>%1:</b><br/>%2")
67  .arg(module->name())
68  .arg(module->config(CSwordModuleInfo::Description));
69  default:
70  return QVariant();
71  }
72 }

99 bool BtBookshelfModel::setData(const QModelIndex & index,
100  const QVariant & value,
101  int role)
102 {
103  int row = index.row();
104  if (role == ModuleHiddenRole
105  && row >= 0
106  && row < m_data.size()
107  && index.column() == 0)
108  {
109  /*
110  Emitting dataChanged here is actually mandatory, but were not doing it
111  directly. Since we're connected to the module, changing its hidden
112  state will emit a signal we catch in moduleHidden(), which in turn is
113  what will actually emit dataChanged().
114  */
115  return m_data.at(row)->setHidden(value.toBool());
116  }
117  return false;
118 }

131 void BtBookshelfModel::addModule(CSwordModuleInfo * const module) {
132  BT_ASSERT(module);
133 
134  if (m_data.contains(module))
135  return;
136 
137  const int index(m_data.size());
138  beginInsertRows(QModelIndex(), index, index);
139  m_data.append(module);
140  BT_CONNECT(module, &CSwordModuleInfo::hiddenChanged,
141  this, &BtBookshelfModel::moduleHidden);
142  BT_CONNECT(module, &CSwordModuleInfo::hasIndexChanged,
143  this, &BtBookshelfModel::moduleIndexed);
144  BT_CONNECT(module, &CSwordModuleInfo::unlockedChanged,
145  this, &BtBookshelfModel::moduleUnlocked);
146  endInsertRows();
147 }

149 void BtBookshelfModel::removeModule(CSwordModuleInfo * const module,
150  bool destroy) {
151  const int index = m_data.indexOf(module);
152  if (index == -1)
153  return;
154 
155  beginRemoveRows(QModelIndex(), index, index);
156  disconnect(module, &CSwordModuleInfo::hiddenChanged,
157  this, &BtBookshelfModel::moduleHidden);
158  disconnect(module, &CSwordModuleInfo::hasIndexChanged,
159  this, &BtBookshelfModel::moduleIndexed);
160  disconnect(module, &CSwordModuleInfo::unlockedChanged,
161  this, &BtBookshelfModel::moduleUnlocked);
162  m_data.removeAt(index);
163  endRemoveRows();
164  if (destroy)
165  delete module;
166 }
