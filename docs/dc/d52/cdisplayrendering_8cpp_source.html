45  }

48  return QString(); //Warning: return already here
49  }

54  linkText = QString();
55  break; //no key is valid for all modules
56

64  case KeyTreeItem::Settings::CompleteShort:
65  if (isBible) {

67  break;
68  }
69  [[fallthrough]];
70 
71  case KeyTreeItem::Settings::ExpandedLong:
72  if (isBible) {
73  linkText = QString("%1 (%2)").arg(vk.key()).arg(module.name());
74  break;
75  }
76  [[fallthrough]];
77 
78  case KeyTreeItem::Settings::CompleteLong:
79  if (isBible) {
80  linkText = vk.key();
81  break;
82  }
83  [[fallthrough]];
84 
85  case KeyTreeItem::Settings::SimpleKey:
86  if (isBible) {
87  if(item.mappedKey() != nullptr) {
88  CSwordVerseKey baseKey(*item.modules().begin());
89  baseKey.setKey(item.key());
90 
91  if (vk.bookName() != baseKey.bookName()) {
92  linkText = vk.shortText();
93  } else if (vk.chapter() != baseKey.chapter()) {
94  linkText = QString("%1:%2").arg(vk.chapter()).arg(vk.verse());
95  } else {
96  linkText = QString::number(vk.verse());
97  }
98 
99  if(vk.isBoundSet()) {
100  linkText += "-";
101  auto const upper = vk.upperBound();
102  auto const lower = vk.lowerBound();
103  if (upper.book() != lower.book()) {
104  linkText += upper.shortText();
105  } else if(upper.chapter() != lower.chapter()) {
106  linkText += QString("%1:%2").arg(upper.chapter())
107  .arg(lower.verse());
108  } else {
109  linkText += QString::number(upper.verse());
110  }
111  }
112  } else {
113  linkText = QString::number(vk.verse());
114  }
115  break;
116  } // else fall through for non-Bible modules
117  [[fallthrough]];
118 
119  default: //default behaviour to return the passed key
120  linkText = item.key();
121  break;
122 
123  }
124 
125 
126  if (linkText.isEmpty()) {
127  return QString("<a name=\"").append(keyToHTMLAnchor(item.key())).append("\"></a>");
128  }
129  else {
130  return QString("<a name=\"").append(keyToHTMLAnchor(item.key())).append("\" ")
131  .append("href=\"")
132  .append(ReferenceManager::encodeHyperlink(module, item.key()))
133  .append("\">").append(linkText).append("</a>\n");
