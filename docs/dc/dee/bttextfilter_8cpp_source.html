121 int BtTextFilter::rewriteHref(int i, const QString& part) {
122  QRegExp rx1("<a\\s(\\w+)=\"([\\s\\S]*)\"\\s(\\w+)=\"([\\s\\S]*)\"");
123  rx1.setMinimal(false);
124  int pos1 = rx1.indexIn(part);
125  if (pos1 >= 0 && rx1.captureCount() == 4) {
126  QString newEntry;
127  if (rx1.cap(1) == "href")
128  newEntry = "<a " + rx1.cap(1) + "=\"" + rx1.cap(2) + "||" + rx1.cap(3) + "=" + rx1.cap(4) + "\" name=\"crossref\">";
129  else
130  newEntry = "<a " + rx1.cap(3) + "=\"" + rx1.cap(4) + "||" + rx1.cap(1) + "=" + rx1.cap(2) + "\" name=\"crossref\">";
131 
132  m_parts[i] = newEntry;
133  }
134  return 1;
135 }
