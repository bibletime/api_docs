53 ListInfoData detectInfo(QString const & data) {
54  ListInfoData list;
55  auto const attrList(data.split("||"));
56 
57  for (auto const & attrPair : attrList) {
58  auto const attr(attrPair.split("="));
59  if (attr.size() == 2) {
60  auto const & attrName = attr[0];
61  auto const & attrValue = attr[1];
62  if (attrName == "note") {
63  list.append(qMakePair(Footnote, attrValue));
64  } else if (attrName == "lemma") {
65  list.append(qMakePair(Lemma, attrValue));
66  } else if (attrName == "morph") {
67  list.append(qMakePair(Morph, attrValue));
68  } else if (attrName == "expansion") {
69  list.append(qMakePair(Abbreviation, attrValue));
70  } else if (attrName == "crossrefs") {
71  list.append(qMakePair(CrossReference, attrValue));
72  } else if (attrName == "href") {
73  list.append(qMakePair(Reference, attrValue));
74  }
75  }
76  }
77  return list;
78 }

81 QString formatInfo(const ListInfoData & list, BtConstModuleList const & modules)
82 {
83  BT_ASSERT(!modules.contains(nullptr) && (modules.size() <= 1 && "not implemented"));
84 
85  if (list.isEmpty())
86  return QString();
87 
88  QString text;
89 
90  for (auto const & infoData : list) {
91  auto const & value = infoData.second;
92  switch (infoData.first) {
93  case Lemma:
94  text.append(decodeStrongs(value));
95  continue;
96  case Morph:
97  text.append(decodeMorph(value));
98  continue;
99  case CrossReference:
100  text.append(decodeCrossReference(value, modules));
101  continue;
102  case Footnote:
103  text.append(decodeFootnote(value));
104  continue;
105  case Abbreviation:
106  text.append(decodeAbbreviation(value));
107  continue;
108  case Text:
109  text.append(value);
110  continue;
111  case Reference:
112  if (value.contains("strongs:")) {
113  auto v(value.right(value.size() - value.lastIndexOf('/')
114  - 1));
115  if (value.contains("GREEK")) {
116  v.prepend('G');
117  } else if (value.contains("HEBREW")) {
118  v.prepend('H');
119  } else {
120  BT_ASSERT(false && "not implemented");
121  }
122  text.append(decodeStrongs(v));
123  } else if (value.contains("sword:")) {
124  text.append(decodeSwordReference(value));
125  continue;
126  } else {
127  BT_ASSERT(false); /// \todo Why is this here?
128  }
129  [[fallthrough]];
130  default:
131  continue;
132  }
133  }
134 
135  return formatInfo(text);
136 }

165 QString decodeCrossReference(QString const & data, BtConstModuleList const & modules) {
166  if (data.isEmpty())
167  return QString("<div class=\"crossrefinfo\"><h3>%1</h3></div>")
168  .arg(QObject::tr("Cross references"));
169 
170  // qWarning("setting crossref %s", data.latin1());
171 
172  DisplayOptions dispOpts;
173  dispOpts.lineBreaks = false;
174  dispOpts.verseNumbers = true;
175 
176  FilterOptions filterOpts;
177  CrossRefRendering renderer(dispOpts, filterOpts);
178  CTextRendering::KeyTree tree;
179 
180  // const bool isBible = true;
181  const CSwordModuleInfo * module(nullptr);
182 
183  // a prefixed module gives the module to look into
184  QRegExp re("^[^ ]+:");
185  // re.setMinimal(true);
186  int pos = re.indexIn(data);
187  if (pos != -1)
188  pos += re.matchedLength() - 1;
189 
190  if (pos > 0) {
191  const QString moduleName = data.left(pos);
192  // qWarning("found module %s", moduleName.latin1());
193  module = CSwordBackend::instance()->findModuleByName(moduleName);
194  }
195 
196  if (!module)
197  module = btConfig().getDefaultSwordModuleByType("standardBible");
198 
199  if (!module && modules.size() > 0)
200  module = modules.at(0);
201 
202  // BT_ASSERT(module); // why? the existense of the module is tested later
203  CTextRendering::KeyTreeItem::Settings settings{
204  false,
205  CTextRendering::KeyTreeItem::Settings::CompleteShort
206  };
207 
208  if (module && (module->type() == CSwordModuleInfo::Bible)) {
209  sword::VerseKey vk;
210  sword::ListKey refs = vk.parseVerseList((const char*) data.mid((pos == -1) ? 0 : pos + 1).toUtf8(), "Gen 1:1", true);
211 
212  for (int i = 0; i < refs.getCount(); i++) {
213  sword::SWKey * const key = refs.getElement(i);
214  BT_ASSERT(key);
215  sword::VerseKey * const vk = dynamic_cast<sword::VerseKey*>(key);
216 
217  if (vk && vk->isBoundSet()) { // render a range of keys
218  tree.emplace_back(
219  QString::fromUtf8(vk->getLowerBound().getText()),
220  QString::fromUtf8(vk->getUpperBound().getText()),
221  module,
222  settings);
223  } else {
224  tree.emplace_back(QString::fromUtf8(key->getText()),
225  QString::fromUtf8(key->getText()),
226  module,
227  settings);
228  }
229  }
230  } else if (module) {
231  tree.emplace_back(data.mid((pos == -1) ? 0 : pos + 1),
232  module,
233  settings);
234  }
235 
236  // qWarning("rendered the tree: %s", renderer.renderKeyTree(tree).latin1());
237  // spanns containing rtl text need dir=rtl on their parent tag to be aligned properly
238  QString lang = "en"; // default english
239  if (module)
240  lang = module->language()->abbrev();
241 
242  return QString("<div class=\"crossrefinfo\" lang=\"%1\"><h3>%2</h3><div class=\"para\" dir=\"%3\">%4</div></div>")
243  .arg(lang)
244  .arg(QObject::tr("Cross references"))
245  .arg(module ? ((module->textDirection() == CSwordModuleInfo::LeftToRight) ? "ltr" : "rtl") : "")
246  .arg(renderer.renderKeyTree(tree));
247 }

252 QString decodeFootnote(QString const & data) {
253  QStringList list = data.split("/");
254  BT_ASSERT(list.count() >= 3);
255  if (!list.count())
256  return QString();
257 
258  FilterOptions filterOpts;
259  filterOpts.footnotes = true;
260  CSwordBackend::instance()->setFilterOptions(filterOpts);
261 
262  const QString modulename = list.first();
263  const QString swordFootnote = list.last();
264 
265  // remove the first and the last and then rejoin it to get a key
266  list.pop_back();
267  list.pop_front();
268  const QString keyname = list.join("/");
269 
270  CSwordModuleInfo * const module = CSwordBackend::instance()->findModuleByName(modulename);
271  if (!module)
272  return QString();
273 
274  QSharedPointer<CSwordKey> key(CSwordKey::createInstance(module));
275  key->setKey(keyname);
276  key->renderedText(CSwordKey::ProcessEntryAttributesOnly); // force entryAttributes
277 
278  auto & m = module->module();
279  const char * const note =
280  m.getEntryAttributes()
281  ["Footnote"][swordFootnote.toLatin1().data()]["body"].c_str();
282 
283  QString text = module->isUnicode() ? QString::fromUtf8(note) : QString(note);
284  text = QString::fromUtf8(m.renderText(
285  module->isUnicode()
286  ? static_cast<const char *>(text.toUtf8())
287  : static_cast<const char *>(text.toLatin1())));
288 
289  return QString("<div class=\"footnoteinfo\" lang=\"%1\"><h3>%2</h3><p>%3</p></div>")
290  .arg(module->language()->abbrev())
291  .arg(QObject::tr("Footnote"))
292  .arg(text);
293 }

344 QString decodeMorph(QString const & data) {
345  QStringList morphs = data.split("|");
346  QString ret;
347 
348  for (auto const & morph : morphs) {
349  //qDebug() << "CInfoDisplay::decodeMorph, morph: " << morph;
350  CSwordModuleInfo * module = nullptr;
351  bool skipFirstChar = false;
352  QString value = "";
353  QString valueClass = "";
354 
355  int valStart = morph.indexOf(':');
356  if (valStart > -1) {
357  valueClass = morph.mid(0, valStart);
358  // qDebug() << "valueClass: " << valueClass;
359  module = CSwordBackend::instance()->findModuleByName(valueClass);
360  }
361  value = morph.mid(valStart + 1); //works for prepended module and without (-1 +1 == 0).
362 
363  // if we don't have a class assigned or desired one isn't installed...
364  if (!module) {
365  // Morphs usually don't have [GH] prepended, but some old OLB
366  // codes do. We should check if we're digit after first char
367  // to better guess this.
368  if (value.size() > 1 && value.at(1).isDigit()) {
369  switch (value.at(0).toLatin1()) {
370  case 'G':
371  module = btConfig().getDefaultSwordModuleByType("standardGreekMorphLexicon");
372  skipFirstChar = true;
373  break;
374  case 'H':
375  module = btConfig().getDefaultSwordModuleByType("standardHebrewMorphLexicon");
376  skipFirstChar = true;
377  break;
378  default:
379  skipFirstChar = false;
380  /// \todo we can't tell here if it's a greek or hebrew moprh code, that's a problem we have to solve
381  // module = getBtConfig().getDefaultSwordModuleByType("standardGreekMorphLexicon");
382  break;
383  }
384  }
385  //if it is still not set use the default
386  if (!module)
387  module = btConfig().getDefaultSwordModuleByType("standardGreekMorphLexicon");
388  }
389 
390  QString text;
391  // BT_ASSERT(module);
392  if (module) {
393  QSharedPointer<CSwordKey> key(CSwordKey::createInstance(module));
394 
395  // skip H or G (language sign) if we have to skip it
396  const bool isOk = key->setKey(skipFirstChar ? value.mid(1) : value);
397  // BT_ASSERT(isOk);
398  if (!isOk) { // try to use the other morph lexicon, because this one failed with the current morph code
399  key->setModule(btConfig().getDefaultSwordModuleByType("standardHebrewMorphLexicon")); /// \todo: what if the module doesn't exist?
400  key->setKey(skipFirstChar ? value.mid(1) : value);
401  }
402 
403  text = key->renderedText();
404  }
405 
406  // if the module wasn't found just display an empty morph info
407  QString lang = "en"; // default to english
408  if (module)
409  lang = module->language()->abbrev();
410  ret.append(QString("<div class=\"morphinfo\" lang=\"%1\"><h3>%2: %3</h3><p>%4</p></div>")
411  .arg(lang)
412  .arg(QObject::tr("Morphology"))
413  .arg(value)
414  .arg(text)
415  );
416  }
417 
418  return ret;
419 }

421 QString decodeSwordReference(QString const & data) {
422  QRegExp rx("sword://(bible|lexicon)/(.*)/(.*)", Qt::CaseInsensitive);
423  rx.setMinimal(false);
424  if (rx.indexIn(data) >= 0) {
425  if (auto * const module =
426  CSwordBackend::instance()->findModuleByName(rx.cap(2)))
427  {
428  std::unique_ptr<CSwordKey> key(CSwordKey::createInstance(module));
429  auto reference = rx.cap(3);
430  key->setKey(reference);
431  return QString("<div class=\"crossrefinfo\" lang=\"%1\">"
432  "<h3>%2</h3><p>%3</p></div>")
433  .arg(module->language()->abbrev())
434  .arg(std::move(reference))
435  .arg(key->renderedText());
436  }
437  }
438  return {};
439 }
