940 void BibleTime::slotDebugTimeout() {
941  std::lock_guard const guard(m_debugWindowLock);
942  if (!m_debugWindow || m_debugWindow->isVisible() == false)
943  return;
944  QTimer::singleShot(0, this, &BibleTime::slotDebugTimeout);
945  if (QObject const * w = QApplication::widgetAt(QCursor::pos())) {
946  QString objectHierarchy;
947  do {
948  QMetaObject const * m = w->metaObject();
949  QString classHierarchy;
950  do {
951  if (!classHierarchy.isEmpty())
952  classHierarchy += ": ";
953  classHierarchy += m->className();
954  m = m->superClass();
955  } while (m);
956  if (!objectHierarchy.isEmpty()) {
957  objectHierarchy += "<br/><b>child of:</b> ";
958  } else {
959  objectHierarchy += "<b>This widget is:</b> ";
960  }
961  objectHierarchy += classHierarchy;
962  w = w->parent();
963  } while (w);
964  m_debugWindow->setText(objectHierarchy);
965  } else {
966  m_debugWindow->setText("No widget");
967  }
