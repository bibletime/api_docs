47 void myMessageOutput(QtMsgType type,
48  QMessageLogContext const &,
49  QString const & message)
50 {
51  QByteArray msg = message.toLatin1();
52  switch (type) {
53  case QtDebugMsg:
54  if (btApp->debugMode()) { // Only show messages if they are enabled!
55  debugStream->write("(BibleTime " BT_VERSION ") Debug: ");
56  debugStream->write(msg);
57  debugStream->write("\n");
58  debugStream->flush();
59  }
60  break;
61  case QtInfoMsg:
62  debugStream->write("(BibleTime " BT_VERSION ") INFO: ");
63  debugStream->write(msg);
64  debugStream->write("\n");
65  debugStream->flush();
66  break;
67  case QtWarningMsg:
68  debugStream->write("(BibleTime " BT_VERSION ") WARNING: ");
69  debugStream->write(msg);
70  debugStream->write("\n");
71  debugStream->flush();
72  break;
73  case QtCriticalMsg:
74  debugStream->write("(BibleTime " BT_VERSION ") CRITICAL: ");
75  debugStream->write(msg);
76  debugStream->write("\nPlease report this bug at "
77  "https://github.com/bibletime/bibletime/issues"
78  "\n");
79  debugStream->flush();
80  break;
81  case QtFatalMsg:
82  debugStream->write("(BibleTime " BT_VERSION ") FATAL: ");
83  debugStream->write(msg);
84  debugStream->write("\nPlease report this bug at "
85  "https://github.com/bibletime/bibletime/issues"
86  "\n");
87 
88  // Dump core on purpose (see qInstallMsgHandler documentation):
89  debugStream->close();
90  std::abort();
91  }
92 }

152 bool BibleTimeApp::initBtConfig() {
153  BT_ASSERT(m_init);
154 
155  BtConfig::InitState const r = BtConfig::initBtConfig();
156  if (r == BtConfig::INIT_OK)
157  return true;
158  if (r == BtConfig::INIT_NEED_UNIMPLEMENTED_FORWARD_MIGRATE) {
159  /// \todo Migrate from btConfigOldApi to BTCONFIG_API_VERSION
160  qWarning() << "BibleTime configuration migration is not yet implemented!!!";
161  if (message::showWarning(
162  nullptr,
163  tr("Warning!"),
164  tr("Migration to the new configuration system is not yet "
165  "implemented. Proceeding might result in <b>loss of data"
166  "</b>. Please backup your configuration files before "
167  "you continue!<br/><br/>Do you want to continue? Press "
168  "\"No\" to quit BibleTime immediately."),
169  QMessageBox::Yes | QMessageBox::No,
170  QMessageBox::No) == QMessageBox::No)
171  return false;
172  } else {
173  BT_ASSERT(r == BtConfig::INIT_NEED_UNIMPLEMENTED_BACKWARD_MIGRATE);
174  if (message::showWarning(
175  nullptr,
176  tr("Error loading configuration!"),
177  tr("Failed to load BibleTime's configuration, because it "
178  "appears that the configuration file corresponds to a "
179  "newer version of BibleTime. This is likely caused by "
180  "BibleTime being downgraded. Loading the new "
181  "configuration file may result in <b>loss of data</b>."
182  "<br/><br/>Do you still want to try to load the new "
183  "configuration file? Press \"No\" to quit BibleTime "
184  "immediately."),
185  QMessageBox::Yes | QMessageBox::No,
186  QMessageBox::No) == QMessageBox::No)
187  return false;
188  }
189  BtConfig::forceMigrate();
190  return true;
191 }
