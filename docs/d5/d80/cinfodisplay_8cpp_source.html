41 CInfoDisplay::CInfoDisplay(BibleTime * parent)
42  : QWidget(parent)
43  , m_mainWindow(parent)
44 {
45  QVBoxLayout * const layout = new QVBoxLayout(this);
46  layout->setContentsMargins(2, 2, 2, 2); // Leave small border
47  setSizePolicy(QSizePolicy::Preferred, QSizePolicy::Preferred);
48  m_textBrowser = new BtTextBrowser(this);
49  m_textBrowser->setContextMenuPolicy(Qt::CustomContextMenu);
50  BT_CONNECT(m_textBrowser, &QTextBrowser::customContextMenuRequested,
51  [this]{
52  QAction selectAllAction(tr("Select all"));
53  selectAllAction.setShortcut(QKeySequence::SelectAll);
54  BT_CONNECT(&selectAllAction, &QAction::triggered,
55  m_textBrowser, &QTextBrowser::selectAll);
56 
57  QAction copyAction(tr("Copy"));
58  copyAction.setShortcut(QKeySequence(Qt::CTRL + Qt::Key_C));
59  BT_CONNECT(&copyAction, &QAction::triggered,
60  m_textBrowser, &QTextBrowser::copy);
61 
62  QMenu menu;
63  menu.addAction(&selectAllAction);
64  menu.addAction(&copyAction);
65  menu.exec(QCursor::pos());
66  });
67  BT_CONNECT(m_textBrowser, &QTextBrowser::anchorClicked,
68  [this](QUrl const & url) {
69  auto const decodedLink(
70  ReferenceManager::decodeHyperlink(
71  url.toString()));
72  if (!decodedLink && !decodedLink->module)
73  return;
74 
75  auto const * const m = decodedLink->module;
76  std::unique_ptr<CSwordKey> key(CSwordKey::createInstance(m));
77  key->setKey(decodedLink->key);
78 
79  setInfo(key->renderedText(), m->language()->abbrev());
80  });
81  layout->addWidget(m_textBrowser);
82  unsetInfo();
83 }

111 void CInfoDisplay::setInfo(const Rendering::ListInfoData & list) {
112  // If the widget is hidden it would be inefficient to render and display the data
