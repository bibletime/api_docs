36 QString CrossRefRendering::entryLink(KeyTreeItem const & item,
37  CSwordModuleInfo const & module)
38 {
39  QString linkText;
40 
41  const bool isBible = (module.type() == CSwordModuleInfo::Bible);
42  CSwordVerseKey vk(&module); // only valid for bible modules, i.e. isBible == true
43  if (isBible) {
44  vk.setKey(item.key());
45  }
46 
47  switch (item.settings().keyRenderingFace) {
48  case KeyTreeItem::Settings::NoKey:
49  linkText = QString();
50  break; //no key is valid for all modules
51  case KeyTreeItem::Settings::CompleteShort:
52  if (isBible) {
53  linkText = vk.shortText();
54  break;
55  }
56  //fall through for non-Bible modules
57  [[fallthrough]];
58  case KeyTreeItem::Settings::CompleteLong:
59  if (isBible) {
60  linkText = vk.key();
61  break;
62  }
63  //fall through for non-Bible modules
64  [[fallthrough]];
65  case KeyTreeItem::Settings::SimpleKey:
66  if (isBible) {
67  linkText = QString::number(vk.verse());
68  break;
69  }
70  //fall through for non-Bible modules
71  [[fallthrough]];
72  default: { //default behaviour to return the passed key
73  linkText = item.key();
74  break;
75  }
76  }
77 
78  if (!linkText.isEmpty()) { //if we have a valid link text
79  // qWarning("rendering");
80  return QString("<a href=\"%1\">%2</a>")
81  .arg(ReferenceManager::encodeHyperlink(module, item.key()))
82  .arg(linkText);
83  }
84 
85  return QString();
86 }
