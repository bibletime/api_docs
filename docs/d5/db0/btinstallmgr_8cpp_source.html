37 template <typename T>
38 inline int calculateIntPercentage(T done, T total) {
39  BT_ASSERT(done >= 0);
40  BT_ASSERT(total >= 0);
41 
42  // Special care (see warning in BtInstallMgr::statusUpdate()).
43  if (done > total)
44  done = total;
45  if (total <= 0.0)
46  return 100;
47 
48  return normalizeCompletionPercentage<int>((done / total) * 100);
49 }
