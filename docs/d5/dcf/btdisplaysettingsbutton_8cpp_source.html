55 m_filterOptions.hebrewCantillation = checked;

57 } else if (action == m_headingsAction) {
58 m_filterOptions.headings = checked;

60 } else if (action == m_morphSegmentationAction) {
61 m_filterOptions.morphSegmentation = checked;
62 Q_EMIT sigFilterOptionsChanged(m_filterOptions);

64 m_filterOptions.scriptureReferences = checked;

66 } else if (action == m_footnotesAction) {
67 m_filterOptions.footnotes = checked;
68 Q_EMIT sigFilterOptionsChanged(m_filterOptions);
69 } else if (action == m_redWordsAction) {
70 m_filterOptions.redLetterWords = checked;
71 Q_EMIT sigFilterOptionsChanged(m_filterOptions);
72 } else {
73 BT_ASSERT(false && "Shouldn't happen!");
74 return;
75 }
76
77 Q_EMIT sigChanged();
78 });

177 addMenuEntry(m_verseNumbersAction, m_displayOptions.verseNumbers);
