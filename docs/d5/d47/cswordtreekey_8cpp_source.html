95void CSwordTreeKey::setModule(const CSwordModuleInfo *newModule) {
96 BT_ASSERT(newModule);

98 BT_ASSERT(newModule->type() == CSwordModuleInfo::GenericBook);
99 BT_ASSERT(dynamic_cast<CSwordBookModuleInfo const *>(newModule));

103 const QString oldKey = key();

105 m_key.copyFrom(

110 //set the key to the root node

112 positionToFirstChild();
