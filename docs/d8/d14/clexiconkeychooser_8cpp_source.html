114 /** Reimplementation. */
115 void CLexiconKeyChooser::refreshContent() {
116  if (m_modules.count() == 1) {
117  m_widget->reset(&m_modules.first()->entries(), 0, true);
118  // qWarning("resetted");
119  }
120  else {
121  std::multimap<unsigned int, QStringList const *> entryMap;
122  for (auto const * const modulePtr : m_modules) {
123  auto const & entries = modulePtr->entries();
124  entryMap.emplace(entries.count(), &entries);
125  }
126 
127  QStringList goodEntries; //The string list which contains the entries which are available in all modules
128 
129  auto it(entryMap.begin()); // iterator to go though all selected modules
130  QStringList refEntries = *(it->second); //copy the items for the first time
131  const QStringList *cmpEntries = (++it)->second; //list for comparision, starts with the second module in the map
132 
133  // Testing for refEntries being empty is not needed for the set union
134  // of all keys, but is a good idea since it is being updated in the
135  // loop. It is necessary for set intersection and prevents a crash.
136  while (it != entryMap.end() && (refEntries.begin() != refEntries.end())) {
137  std::set_union(
138  refEntries.begin(), --(refEntries.end()), //--end() is the last valid entry
139  cmpEntries->begin(), --(cmpEntries->end()),
140  std::back_inserter(goodEntries) //append valid entries to the end of goodEntries
141  );
142 
143  cmpEntries = ( ++it )->second; //this is a pointer to the string list of a new module
144 
145  /*
146  * use the good entries for next comparision,
147  * because the final list can only have the entries of goodEntries as maxiumum
148  */
149  refEntries = goodEntries;
150  }
151 
152  m_widget->reset(goodEntries, 0, true); //write down the entries
153  } //end of ELSE
154 
155 }

157 void CLexiconKeyChooser::setModules(const BtConstModuleList &modules,
158  bool refresh)
159 {
160  using CSLMI = CSwordLexiconModuleInfo;
161 
162  while (!m_modules.isEmpty())
163  m_modules.takeFirst(); // not deleting the pointer
164 
165  for (auto const * const m : modules)
166  if (CSLMI const * const lexicon = dynamic_cast<CSLMI const *>(m))
167  m_modules.append(lexicon);
168 
169  if (refresh) {
170  refreshContent();
171  }
