218 sword::InstallSource CSwordSetupInstallSourcesDialog::getSource() {
219  sword::InstallSource newSource(""); //empty, invalid Source
220  if (this->isRemote(m_protocolCombo->currentText())) {
221  if (m_protocolCombo->currentText() == PROTO_FTP) {
222  newSource.type = "FTP";
223  }
224  else if (m_protocolCombo->currentText() == PROTO_SFTP) {
225  newSource.type = "SFTP";
226  }
227  else if (m_protocolCombo->currentText() == PROTO_HTTP) {
228  newSource.type = "HTTP";
229  }
230  else if (m_protocolCombo->currentText() == PROTO_HTTPS) {
231  newSource.type = "HTTPS";
232  }
233  newSource.source = m_serverEdit->text().toUtf8();
234  //a message to the user would be nice, but we're in message freeze right now (1.5.1)
235  if (m_serverEdit->text().right(1) == "/") { //remove a trailing slash
236  newSource.source = m_serverEdit->text().mid(0, m_serverEdit->text().length() - 1).toUtf8();
237  }
238  }
239  else {
240  newSource.type = "DIR";
241  newSource.source = "local";
242  }
243  newSource.caption = m_captionEdit->text().toUtf8();
244  newSource.directory = m_pathEdit->text().toUtf8();
245  newSource.uid = newSource.source;
246 
247  return newSource;
248 }
