175 template <class T>
176 QModelIndex getGroup(CSwordModuleInfo const & module,
177 QModelIndex const parentIndex,
178 bool & beginInsert)
179 {
180 BookshelfModel::Item & parentItem = getItem(parentIndex);
181 int groupIndex;
182 T * groupItem = parentItem.getGroupItem<T>(module, groupIndex);
183
184 if (!groupItem) {
185 groupItem = new T(module);
186 groupIndex = parentItem.indexFor(*groupItem);
187 if (beginInsert) {
188 beginInsertRows(parentIndex, groupIndex, groupIndex);
189 beginInsert = false;
190 }
191 parentItem.insertChild(groupIndex, groupItem);
192 }
193 return index(groupIndex, 0, parentIndex);
194 }
