74 void CBookTreeChooser::setKey(CSwordKey* newKey, const bool emitSignal) {
75 
76  if (m_key != newKey ) {
77  m_key = dynamic_cast<CSwordTreeKey*>(newKey);
78  }
79 
80  const QString key = m_key->key(); //key as text, path
81 
82  QTreeWidgetItem* matching_item = m_treeView->topLevelItem(0);
83 
84  QTreeWidgetItemIterator it(m_treeView);
85  while (*it) {
86  if ((*it)->text(1) == key) {
87  matching_item = (*it);
88  break;
89  }
90  ++it;
91  }
92 
93  m_treeView->setCurrentItem( matching_item );
94  m_treeView->scrollToItem(matching_item);
95 
96  if (emitSignal) {
97  Q_EMIT keyChanged(m_key);
98  }
99 }

101 void CBookTreeChooser::setModules(const BtConstModuleList &modules,
102  bool refresh)
103 {
104  using CSBMI = CSwordBookModuleInfo;
105 
106  //Add given modules into private list
107  m_modules.clear();
108  for (auto const * const m : modules) {
109  const CSBMI *book = dynamic_cast<const CSBMI*>(m);
110  if (book != nullptr) {
111  m_modules.append(book);
112  }
113  }
114 
115  //if there exists a module and a key, setup the visible tree
116  if (refresh && m_modules.count() && m_key) {
117  auto const offset = m_key->offset(); //actually unnecessary, taken care of in setupTree
118  setupTree();
119  m_key->setOffset( offset );
120 
121  adjustFont(); //only when refresh is set.
122  }
123 }

183 /** Populates tree widget with items. */
184 void CBookTreeChooser::addKeyChildren(CSwordTreeKey* key, QTreeWidgetItem* item) {
185  if (key->hasChildren()) {
186  key->positionToFirstChild();
187  do {
188  QStringList columns;
189  columns << key->getLocalNameUnicode() << key->key();
190  QTreeWidgetItem *i = new QTreeWidgetItem(item, columns, QTreeWidgetItem::Type);
191  i->setData(0, Qt::ToolTipRole, key->getLocalNameUnicode());
192  auto const offset = key->offset();
193  addKeyChildren(key, i);
194  key->setOffset(offset);
195  }
196  while (key->positionToNextSibling());
197  }
198 }
