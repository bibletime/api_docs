57 QString CSwordKey::renderedText(const CSwordKey::TextRenderType mode) {
58  BT_ASSERT(m_module);
59 
60  auto & m = m_module->module();
61  sword::VerseKey * vk_mod = dynamic_cast<sword::VerseKey *>(m.getKey());
62  if (vk_mod)
63  vk_mod->setIntros(true);
64 
65  m.getKey()->setText(rawKey());
66 
67  if (m_module->type() == CSwordModuleInfo::Lexicon) {
68  m_module->snap();
69  /* In lexicons make sure that our key (e.g. 123) was successfully set to the module,
70  i.e. the module key contains this key (e.g. 0123 contains 123) */
71 
72  if (sword::stricmp(m.getKey()->getText(), rawKey())
73  && !strstr(m.getKey()->getText(), rawKey()))
74  {
75  qDebug("return an empty key for %s", m.getKey()->getText());
76  return QString();
77  }
78  }
79 
80  if (key().isNull())
81  return QString();
82 
83  bool DoRender = mode != ProcessEntryAttributesOnly;
84  QString text = QString::fromUtf8(m.renderText(nullptr, -1, DoRender));
85  if (!DoRender)
86  return QString();
87 
88  // This is yucky, but if we want strong lexicon refs we have to do it here.
89  if (m_module->type() == CSwordModuleInfo::Lexicon) {
90  const QString t(text);
91  const QRegExp rx("(GREEK|HEBREW) for 0*([1-9]\\d*)"); // ignore 0's before number
92  int pos = 0;
93  while ((pos = rx.indexIn(t, pos)) != -1) {
94  const QString language = rx.cap(1);
95  const QString langcode = QString(language.at(0)); // "G" or "H"
96  const QString number = rx.cap(2);
97  const QString paddednumber = number.rightJustified(5, '0'); // Form 00123
98 
99  text.replace(
100  QRegExp(QString(
101  "(>[^<>]+)" // Avoid replacing inside tags
102  "\\b(0*%1)\\b").arg(number)), // And span around 0's
103  QString("\\1<span lemma=\"%1%2\"><a href=\"strongs://%3/%4\">\\2</a></span>")
104  .arg(langcode, paddednumber, language, paddednumber)
105  );
106  pos += rx.matchedLength();
107  }
108  }
109 
110  if (mode == HTMLEscaped) {
111  /*
112  Here we encode all non-latin1 characters as HTML unicode entities
113  in the form &#<decimal unicode value here>;
114  */
115  QString ret;
116 
117  // Reserve characters to reduce number of memory allocations:
118  ret.reserve(text.size());
119 
120  for (auto const & c : text) {
121  if (c.toLatin1()) {
122  ret.append(c);
123  } else {
124  ret.append("&#").append(c.unicode()).append(";");
125  }
126  }
127 
128  return ret;
129  }
130  else {
131  return text;
132  }
133 }
