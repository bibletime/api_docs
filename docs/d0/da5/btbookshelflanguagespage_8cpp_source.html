74 void BtBookshelfLanguagesPage::initializePage() {

76  { // Select languages:

87 
88  bool scrolledToFirstSelected = false;
89  for (int row = 0; row < m_model->rowCount(); ++row) {
90  QStandardItem * const item = m_model->item(row, 0);
91  if (languages.contains(item->text())) {
92  item->setCheckState(Qt::Checked);
93  // Scroll to first selected item:
94  if (!scrolledToFirstSelected) {
95  m_languagesListView->scrollTo(m_model->indexFromItem(item));
96  scrolledToFirstSelected = true;
97  }
98  }
99  }
100  }
101 
102  retranslateUi();
103 }

105 void BtBookshelfLanguagesPage::initializeLanguages() {
106  // Get languages from sources:
107  std::set<QString> languages;
108  for (auto const & sourceName : btWizard().selectedSources()) {
109  std::shared_ptr<CSwordBackend const> const backend(
110  BtInstallBackend::backend(
111  BtInstallBackend::source(sourceName)));
112  for (auto const * module : backend->moduleList())
113  languages.insert(module->language()->translatedName());
114  }
115 
116  // Update languages model:
117  m_model->clear();
118  for (auto const & lang : languages)
119  m_model->appendItem(lang);
120  if (languages.size() == 1u)
121  m_model->item(0, 0)->setCheckState(Qt::Checked);
122 }
