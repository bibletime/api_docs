49  bool filterAcceptsRow(int sourceRow,
50  QModelIndex const & sourceParentIndex)
51  const final override
52  {
53  auto const & model = *sourceModel();
54  auto const itemIndex = model.index(sourceRow, 0, sourceParentIndex);
55 
56  // Do recursive hidden check, if configured:
57  if (!m_showHidden
58  && model.data(itemIndex,
59  BtBookshelfModel::ModuleHiddenRole).toBool())
60  return false;
61 
62  // Do other recursive checks:
63  return filterAcceptsRowNoHiddenCheck(sourceRow, sourceParentIndex);
64  }
65 
66  bool filterAcceptsRowNoHiddenCheck(int sourceRow,
67  QModelIndex const & sourceParentIndex)
68  const
69  {
70  auto const & model =
71  *static_cast<BtBookshelfTreeModel *>(sourceModel());
72  auto const itemIndex = model.index(sourceRow, 0, sourceParentIndex);
73 
74  // Accept subtrees if it has any accepted children:
75  if (auto const numRows = model.rowCount(itemIndex)) {
76  for (int i = 0; i < numRows; ++i)
77  if (filterAcceptsRowNoHiddenCheck(i, itemIndex))
78  return true;
79  return false;
80  }
81 
82  auto const * const module = model.module(itemIndex);
83  BT_ASSERT(module);
84  return (module->type() == m_moduleType)
85  || ((m_moduleType == CSwordModuleInfo::Bible)
86  && (module->type() == CSwordModuleInfo::Commentary));
87  }

146 QAction * BtModuleChooserMenu::newAction(QMenu * parentMenu,
147  QModelIndex const & itemIndex)
148 {
149  auto * const action = BtMenuView::newAction(parentMenu, itemIndex);
150 
151  auto const & sortedModel = *static_cast<SortModel *>(m_sortedModel);
152 
153  auto const sourceItemIndex = sortedModel.mapToSource(itemIndex);
154  auto const * const module =
155  sortedModel.m_sourceModel->module(sourceItemIndex);
156  BT_ASSERT(module);
157  action->setCheckable(true);
158  auto const & moduleName = module->name();
159  action->setChecked(moduleName == m_selectedModule);
160 
161  if (m_flags & DisableSelectedModule) {
162  action->setDisabled(moduleName == m_selectedModule
163  || m_newModulesToUse.contains(moduleName)
164  || module->isLocked());
165  } else {
166  action->setDisabled((moduleName != m_selectedModule
167  && m_newModulesToUse.contains(moduleName))
168  || module->isLocked());
169  }
170 
171  // Disable non-Bible modules on first button:
172  if ((m_flags & DisableNonBiblesOnFirstButton)
173  && m_buttonIndex <= 0
174  && sortedModel.m_moduleType == CSwordModuleInfo::Bible
175  && module->category() != CSwordModuleInfo::Bibles)
176  action->setDisabled(true);
177 
178  return action;
179 }
