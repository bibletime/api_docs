38 /**
39 * This class is used to keep track of the text strongs results.
40 * It only keeps track of one instance of a strongs text result.
41 *
42 * The functions of the class are:
43 * - Store an instance of a strongs text result.
44 * - Each strongs text result will contain a list of verses (keyNames).
45 * - The number of verses (keyNames) is returned by keyCount().
46 * - The text for the strongs text result is returned by keyText().
47 * - The list of verses (keyNames) is returned by getKeyList() [as QStringList].
48 *
49 * To add a new verse to a strongs text result use addKeyName.
50 */

52  public: /* Methods: */
53  StrongsResult() {}
54  StrongsResult(QString const & text,
55  QString const & keyName)
56  : m_text(text)
57  {
58  m_keyNameList.append(keyName);
59  }
60 
61  const QString &keyText() const { return m_text; }
62  int keyCount() const { return m_keyNameList.count(); }
63  void addKeyName(QString const & keyName) {
64  if (m_keyNameList.contains(keyName)) return;
65  m_keyNameList.append(keyName);
66  }
67 
68  QStringList const & getKeyList() const { return m_keyNameList; }
