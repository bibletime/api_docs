45void CScrollButton::mouseMoveEvent(QMouseEvent *e) {
46 if (m_isLocked) {
47 // Recalculate the center of the widget (might change during grab):
48 QPoint center(mapToGlobal(QPoint(width() / 2, height() / 2)));
49
50 // Calculate movement change:
51 int const vchange = (e->globalPosition().toPoint().y() - center.y());
52 if (vchange != 0) {
53 // Adapt the change value, so we get a more natural feeling:
54 if(vchange > 0)
55 m_movement += pow(vchange/10.0f, 1.2);
56 else // (vchange < 0)
57 m_movement -= pow(-vchange/10.0f, 1.2);
58
59 // Emit the change request signal only when the mouse was moved far enough
60 if (m_movement >= 1.0 || m_movement <= -1.0) {
61 Q_EMIT change_requested(static_cast<int>(m_movement));
62 m_movement = 0.0;
63 }
64 }
65
66 // Move the mouse cursor to the center of this widget:
67 QCursor::setPos(center);
68 }
69 else {
70 QToolButton::mouseMoveEvent(e);
71 }
72}
