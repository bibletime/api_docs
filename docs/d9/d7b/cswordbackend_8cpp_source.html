88 std::shared_ptr<CSwordBackend::AvailableLanguagesCacheContainer const>
89 CSwordBackend::availableLanguages() noexcept
90 {
91  auto oldCache = std::atomic_load_explicit(&m_availableLanguagesCache,
92  std::memory_order_acquire);
93  if (oldCache)
94  return oldCache;
95 
96  auto const generateCache =
97  [&model = *m_dataModel] {
98  AvailableLanguagesCacheContainer newCache;
99  for (auto const * const mod : model.moduleList()) {
100  newCache.emplace(mod->language());
101  if (auto lang2 = mod->glossaryTargetlanguage())
102  newCache.emplace(std::move(lang2));
103  }
104 
105  return std::make_shared<AvailableLanguagesCacheContainer const>(
106  std::move(newCache)); // also makes container const
107  };
108 
109  for (auto newCache = generateCache();; newCache = generateCache())
110  if (std::atomic_compare_exchange_strong_explicit(
111  &m_availableLanguagesCache,
112  &oldCache,
113  newCache,
114  std::memory_order_acq_rel,
115  std::memory_order_relaxed))
116  return newCache;
117 }

119 void CSwordBackend::uninstallModules(BtConstModuleSet const & toBeDeleted) {
120  if (toBeDeleted.empty())
121  return;
122  m_dataModel->removeModules(toBeDeleted);
123  Q_EMIT sigSwordSetupChanged(RemovedModules);
124 
125  BtInstallMgr installMgr;
126  QMap<QString, sword::SWMgr *> mgrDict; // Maps config paths to SWMgr objects
127  for (CSwordModuleInfo const * const mInfo : toBeDeleted) {
128  // Find the install path for the sword manager:
129  QString dataPath = mInfo->config(CSwordModuleInfo::DataPath);
130  if (dataPath.left(2) == "./")
131  dataPath = dataPath.mid(2);
132 
133  QString prefixPath =
134  mInfo->config(CSwordModuleInfo::AbsoluteDataPath) + "/";
135  if (prefixPath.contains(dataPath)) {
136  // Remove module part to get the prefix path:
137  prefixPath = prefixPath.remove(prefixPath.indexOf(dataPath),
138  dataPath.length());
139  } else { // This is an error, should not happen
140  qWarning() << "Removing" << mInfo->name()
141  << "didn't succeed because the absolute path"
142  << prefixPath << "didn't contain the data path"
143  << dataPath;
144  continue; // don't remove this, go to next of the for loop
145  }
146 
147  // Create the sword manager and remove the module
148  sword::SWMgr * mgr = mgrDict[prefixPath];
149  if (!mgr) { // Create new mgr if it's not yet available
150  mgrDict.insert(prefixPath,
151  new sword::SWMgr(prefixPath.toLocal8Bit()));
152  mgr = mgrDict[prefixPath];
153  }
154  qDebug() << "Removing the module" << mInfo->name() << "...";
155  installMgr.removeModule(mgr, mInfo->module().getName());
156  }
157  qDeleteAll(toBeDeleted);
158  qDeleteAll(mgrDict);
159  mgrDict.clear();
160 }

178 CSwordBackend::LoadError CSwordBackend::initModules(const SetupChangedReason reason) {
179  // qWarning("globalSwordConfigPath is %s", globalConfPath);
180 
181  shutdownModules(); // Remove previous modules
182  m_dataModel->clear();
183 
184  const LoadError ret = static_cast<LoadError>(m_manager.load());
185 
186  for (auto const & modulePair : m_manager.getModules()) {
187  sword::SWModule * const curMod = modulePair.second;
188  BT_ASSERT(curMod);
189  std::unique_ptr<CSwordModuleInfo> newModule;
190 
191  const char * const modType = curMod->getType();
192  if (!strcmp(modType, "Biblical Texts")) {
193  newModule = std::make_unique<CSwordBibleModuleInfo>(*curMod, *this);
194  newModule->setDisplay(&m_chapterDisplay);
195  } else if (!strcmp(modType, "Commentaries")) {
196  newModule = std::make_unique<CSwordCommentaryModuleInfo>(*curMod,
197  *this);
198  newModule->setDisplay(&m_entryDisplay);
199  } else if (!strcmp(modType, "Lexicons / Dictionaries")) {
200  newModule = std::make_unique<CSwordLexiconModuleInfo>(*curMod,
201  *this);
202  newModule->setDisplay(&m_entryDisplay);
203  } else if (!strcmp(modType, "Generic Books")) {
204  newModule = std::make_unique<CSwordBookModuleInfo>(*curMod, *this);
205  newModule->setDisplay(&m_bookDisplay);
206  } else {
207  continue;
208  }
209 
210  // Append the new modules to our list, but only if it's supported
211  // The constructor of CSwordModuleInfo prints a warning on stdout
212  if (!newModule->hasVersion()
213  || (newModule->minimumSwordVersion() <= sword::SWVersion::currentVersion))
214  {
215 
216  /* There is currently a deficiency in sword 1.8.1 in that backend->setCipherKey() does
217  * not work correctly for modules from which data was already fetched. Therefore we have to
218  * reload the modules. The cipher key must be set before any read occurs on the module.
219  * Reading from the module can happen in subtle ways. The addModule below causes a read
220  * to determine if the locked or unlocked icon is used by the model.
221  */
222  if (newModule->isEncrypted()) {
223  auto const unlockKey(
224  btConfig().getModuleEncryptionKey(newModule->name()));
225  if (!unlockKey.isNull())
226  m_manager.setCipherKey(
227  newModule->name().toUtf8().constData(),
228  unlockKey.toUtf8().constData());
229  }
230 
231  /// \todo Refactor data model to use shared_ptr to contain works
232  m_dataModel->addModule(newModule.get());
233  newModule.release();
234  }
235  }
236 
237  Q_EMIT sigSwordSetupChanged(reason);
238  return ret;
239 }

241 void CSwordBackend::Private::addRenderFilters(sword::SWModule * module,
242  sword::ConfigEntMap & section)
243 {
244  auto entry(section.find("SourceType"));
245  if (entry != section.end()) {
246  if (entry->second == "OSIS") {
247  module->addRenderFilter(&m_osisFilter);
248  return;
249  } else if (entry->second == "ThML") {
250  module->addRenderFilter(&m_thmlFilter);
251  return;
252  } else if (entry->second == "TEI") {
253  module->addRenderFilter(&m_teiFilter);
254  return;
255  } else if (entry->second == "GBF") {
256  module->addRenderFilter(&m_gbfFilter);
257  return;
258  } else if (entry->second == "PLAIN") {
259  module->addRenderFilter(&m_plainFilter);
260  return;
261  }
262  }
263 
264  // No driver found
265  entry = section.find("ModDrv");
266  if (entry != section.end() && (entry->second == "RawCom" || entry->second == "RawLD"))
267  module->addRenderFilter(&m_plainFilter);

275 void CSwordBackend::Private::shutdownModules() {
276  //BT mods are deleted now, delete Sword mods, too.
277  deleteAllModules();
278 
279  /* Cipher filters must be handled specially, because SWMgr creates them,
280  * stores them in cipherFilters and cleanupFilters and attaches them to locked
281  * modules. If these modules are removed, the filters need to be removed as well,
282  * so that they are re-created for the new module objects.
283  */
284  for (auto const & filterPair : cipherFilters) {
285  //Delete the Filter and remove it from the cleanup list
286  cleanupFilters.remove(filterPair.second);
287  delete filterPair.second;
288  }

507 // Return a list of used Sword dirs. Useful for the installer.
508 QStringList CSwordBackend::swordDirList() const {
509  namespace DU = util::directory;
510 
511  // Get the set of sword directories that could contain modules:
512  QSet<QString> swordDirSet;
513  QStringList configs;
514 
515  if (QFile(getPrivateSwordConfigFile()).exists()) {
516  // Use the private sword.conf file:
517  configs << getPrivateSwordConfigFile();
518  } else {
519  /*
520  Did not find private sword.conf, will use shared sword.conf files to
521  build the private one. Once the private sword.conf exist, the shared
522  ones will not be searched again.
523  */
524  configs = getSharedSwordConfigFiles();
525 
526 #ifdef Q_OS_WIN
527  /*
528  On Windows, add the shared sword directory to the set so the new
529  private sword.conf will have it. The user could decide to delete this
530  shared path and it will not automatically come back.
531  */
532  swordDirSet << DU::convertDirSeparators(qgetenv("SWORD_PATH"));
533 #endif
534  }
535 
536  // Search the sword.conf file(s) for sword directories that could contain modules
537  for (auto const & filename : configs) {
538  if (!QFileInfo(filename).exists())
539  continue;
540 
541  /*
542  Get all DataPath and AugmentPath entries from the config file and add
543  them to the list:
544  */
545  sword::SWConfig conf(filename.toUtf8().constData());
546  swordDirSet << QDir(QTextCodec::codecForLocale()->toUnicode(conf["Install"]["DataPath"].c_str())).absolutePath();
547 
548  const sword::ConfigEntMap group(conf["Install"]);
549  for (auto its = group.equal_range("AugmentPath");
550  its.first != its.second;
551  ++(its.first))
552  {
553  // Added augment path:
554  swordDirSet << QDir(QTextCodec::codecForLocale()->toUnicode(its.first->second.c_str())).absolutePath();
555  }
556  }
557 
558  // Add the private sword path to the set if not there already:
559  swordDirSet << getPrivateSwordConfigPath();
560 
561  QStringList swordDirs;
562  for (auto dir: swordDirSet)
563  swordDirs.append(dir);
564  return swordDirs;
565 }
