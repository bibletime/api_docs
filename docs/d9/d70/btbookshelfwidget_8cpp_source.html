178 bool BtBookshelfWidget::eventFilter(QObject *object, QEvent *event) {
179  BT_ASSERT(object == m_nameFilterEdit);
180  if (event->type() == QEvent::KeyPress) {
181  QKeyEvent *e = static_cast<QKeyEvent*>(event);
182  switch (e->key()) {
183  case Qt::Key_Up:
184  case Qt::Key_Down:
185  case Qt::Key_Enter:
186  case Qt::Key_Return:
187  QApplication::sendEvent(m_treeView, event);
188  return true;
189  default:
190  break;
191  }
192  }
193  return false;
194 }

196 void BtBookshelfWidget::findExpanded(
197  const QModelIndex& index,
198  QString prefix,
199  QStringList * expandedPaths) {
200  QString path = index.data(Qt::DisplayRole).toString();
201  if (m_treeView->isExpanded(index)) {
202  if (! prefix.isEmpty())
203  path = prefix + QString("/") + path;
204  expandedPaths->append(path);
205  }
206  int rows = m_postFilterModel->rowCount(index);
207  for (int row = 0; row < rows; ++row) {
208  auto childIndex = m_postFilterModel->index(row, 0, index);
209  findExpanded(childIndex, path, expandedPaths);
210  }
211 }

213 void BtBookshelfWidget::restoreExpanded(const QModelIndex& index, const QStringList& nodeList) {
214  if (nodeList.isEmpty())
215  return;
216  QString firstNode = nodeList.first();
217  QStringList remainingNodes = nodeList;
218  remainingNodes.removeFirst();
219  int rows = m_postFilterModel->rowCount(index);
220  for (int row = 0; row < rows; ++row) {
221  auto childIndex = m_postFilterModel->index(row, 0, index);
222  QString text = childIndex.data(Qt::DisplayRole).toString();
223  if (firstNode == text) {
224  m_treeView->expand(childIndex);
225  restoreExpanded(childIndex, remainingNodes);
226  break;
227  }
228  }
229 }
