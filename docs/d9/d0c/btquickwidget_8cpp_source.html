52 // Reimplementation from QQuickWidget
53 void BtQuickWidget::dragEnterEvent( QDragEnterEvent* e ) {
54  if ( ! e->mimeData()->hasFormat("BibleTime/Bookmark"))
55  return;
56 
57  const QMimeData* mimedata = e->mimeData();
58  if (mimedata == nullptr)
59  return;
60 
61  const BTMimeData* btmimedata = qobject_cast<const BTMimeData*>(mimedata);
62  if (btmimedata == nullptr)
63  return;
64 
65  BookmarkItem item = (qobject_cast<const BTMimeData*>(e->mimeData()))->bookmark();
66  QString moduleName = item.module();
67  CSwordModuleInfo *m = CSwordBackend::instance()->findModuleByName(moduleName);
68  BT_ASSERT(m);
69 
70  // Is bible reference bookmark compatible with the module type?
71  CSwordModuleInfo::ModuleType bookmarkType = m->type();
72  if ((bookmarkType == CSwordModuleInfo::Bible ||
73  bookmarkType == CSwordModuleInfo::Commentary)) {
74  if (getQmlInterface()->isBibleOrCommentary()) {
75  e->acceptProposedAction();
76  return;
77  }
78  }
79 
80  QQuickWidget::dragEnterEvent(e);
81  return;
82 }

262 void BtQuickWidget::scrollTimerSlot() {
263  QPoint globalPoint = QCursor::pos();
264  QPoint point = mapFromGlobal(globalPoint);
265  int y = point.y();
266  if ((y >= 0) & (y-height() < 0))
267  return;
268  if (y < 0) {
269  scroll(-8);
270  } else {
271  scroll(8);
272  }
273  int y2 = y * y;
274  if (y2 > 100)
275  y2 = 100;
276  int interval = 500 / y2;
277  m_timer.setInterval(interval);
278 }
