54 char ThmlToHtml::processText(sword::SWBuf &buf, const sword::SWKey *key,
55  const sword::SWModule *module)
56 {
57  sword::ThMLHTML::processText(buf, key, module);
58 
59  CSwordModuleInfo* m = CSwordBackend::instance()->findModuleByName( module->getName() );
60 
61  if (m && !(m->has(CSwordModuleInfo::lemmas) || m->has(CSwordModuleInfo::strongNumbers))) { //only parse if the module has strongs or lemmas
62  return 1;
63  }
64 
65  QString result;
66 
67  QString t = QString::fromUtf8(buf.c_str());
68  QRegExp tag("([.,;]?<sync[^>]+(type|value)=\"([^\"]+)\"[^>]+(type|value)=\"([^\"]+)\"([^<]*)>)+");
69 
70  QStringList list;
71  int lastMatchEnd = 0;
72  int pos = tag.indexIn(t, 0);
73 
74  if (pos == -1) { //no strong or morph code found in this text
75  return 1; //WARNING: Return already here
76  }
77 
78  while (pos != -1) {
79  list.append(t.mid(lastMatchEnd, pos + tag.matchedLength() - lastMatchEnd));
80 
81  lastMatchEnd = pos + tag.matchedLength();
82  pos = tag.indexIn(t, pos + tag.matchedLength());
83  }
84 
85  if (!t.right(t.length() - lastMatchEnd).isEmpty()) {
86  list.append(t.right(t.length() - lastMatchEnd));
87  }
88 
89  tag = QRegExp("<sync[^>]+(type|value|class)=\"([^\"]+)\"[^>]+(type|value|class)=\"([^\"]+)\"[^>]+((type|value|class)=\"([^\"]+)\")*([^<]*)>");
90 
91  for (auto e : list) {
92 
93  // pass text ahead of <sync> stright through
94  int pos2 = tag.indexIn(e, 0);
95  if (pos2 >= 0) {
96  result.append(e.left(pos2));
97  e = e.remove(0, pos2);
98  }
99 
100  // parse <sync> and change to <span>
101  bool hasLemmaAttr = false;
102  bool hasMorphAttr = false;
103 
104  int pos = tag.indexIn(e, 0);
105  bool insertedTag = false;
106  QString value;
107  QString valueClass;
108 
109  while (pos != -1) {
110  bool isMorph = false;
111  bool isStrongs = false;
112  value = QString();
113  valueClass = QString();
114 
115  // check 3 attribute/value pairs
116 
117  for (int i = 1; i < 6; i += 2) {
118  if (i > 4)
119  i++;
120 
121  if (tag.cap(i) == "type") {
122  isMorph = (tag.cap(i + 1) == "morph");
123  isStrongs = (tag.cap(i + 1) == "Strongs");
124  }
125  else if (tag.cap(i) == "value") {
126  value = tag.cap(i + 1);
127  }
128  else if (tag.cap(i) == "class") {
129  valueClass = tag.cap(i + 1);
130  }
131  }
132 
133  // prepend the class qualifier to the value
134  if (!valueClass.isEmpty()) {
135  value = valueClass + ":" + value;
136  // value.append(":").append(value);
137  }
138 
139  if (value.isEmpty()) {
140  break;
141  }
142 
143  //insert the span
144  if (!insertedTag) {
145  e.replace(pos, tag.matchedLength(), "</span>");
146  pos += 7;
147 
148  QString rep = QString("<span lemma=\"").append(value).append("\">");
149  int startPos = 0;
150  QChar c = e[startPos];
151 
152  while ((startPos < pos) && (c.isSpace() || c.isPunct())) {
153  ++startPos;
154  c = e[startPos];
155  }
156 
157  hasLemmaAttr = isStrongs;
158  hasMorphAttr = isMorph;
159 
160  e.insert( startPos, rep );
161  pos += rep.length();
162  }
163  else { //add the attribute to the existing tag
164  e.remove(pos, tag.matchedLength());
165 
166  if ((!isMorph && hasLemmaAttr) || (isMorph && hasMorphAttr)) { //we append another attribute value, e.g. 3000 gets 3000|5000
167  //search the existing attribute start
168  QRegExp attrRegExp( isMorph ? "morph=\".+(?=\")" : "lemma=\".+(?=\")" );
169  attrRegExp.setMinimal(true);
170  const int foundAttrPos = e.indexOf(attrRegExp, pos);
171 
172  if (foundAttrPos != -1) {
173  e.insert(foundAttrPos + attrRegExp.matchedLength(), QString("|").append(value));
174  pos += value.length() + 1;
175 
176  hasLemmaAttr = !isMorph;
177  hasMorphAttr = isMorph;
178  }
179  }
180  else { //attribute was not yet inserted
181  const int attrPos = e.indexOf(QRegExp("morph=|lemma="), 0);
182 
183  if (attrPos >= 0) {
184  QString attr;
185  attr.append(isMorph ? "morph" : "lemma").append("=\"").append(value).append("\" ");
186  e.insert(attrPos, attr);
187 
188  hasMorphAttr = isMorph;
189  hasLemmaAttr = !isMorph;
190 
191  pos += attr.length();
192  }
193  }
194  }
195 
196  insertedTag = true;
197  pos = tag.indexIn(e, pos);
198  }
199 
200  result.append( e );
201  }
202 
203  if (list.count())
204  buf = result.toUtf8();
205 
206  return 1;
207 }

210 bool ThmlToHtml::handleToken(sword::SWBuf &buf, const char *token,
211  sword::BasicFilterUserData *userData)
212 {
213  if (!substituteToken(buf, token) && !substituteEscapeString(buf, token)) {
214  sword::XMLTag const tag(token);
215  BT_ASSERT(dynamic_cast<UserData *>(userData));
216  UserData * const myUserData = static_cast<UserData *>(userData);
217  // Hack to be able to call stuff like Lang():
218  sword::SWModule const * const myModule =
219  const_cast<sword::SWModule *>(myUserData->module);
220  char const * const tagName = tag.getName();
221  if (!tagName) // unknown tag, pass through:
222  return sword::ThMLHTML::handleToken(buf, token, userData);
223  if (!sword::stricmp(tagName, "foreign")) {
224  // A text part in another language, we have to set the right font
225 
226  if (const char * const tagLang = tag.getAttribute("lang"))
227  buf.append("<span class=\"foreign\" lang=\"")
228  .append(tagLang)
229  .append("\">");
230  } else if (!sword::stricmp(tagName, "sync")) {
231  // If Morph or Strong or Lemma:
232  if (const char * const tagType = tag.getAttribute("type"))
233  if (!sword::stricmp(tagType, "morph")
234  || !sword::stricmp(tagType, "Strongs")
235  || !sword::stricmp(tagType, "lemma"))
236  buf.append('<').append(token).append('>');
237  } else if (!sword::stricmp(tagName, "note")) { // <note> tag
238  if (!tag.isEmpty()) {
239  if (!tag.isEndTag()) {
240  buf.append(" <span class=\"footnote\" note=\"")
241  .append(myModule->getName())
242  .append('/')
243  .append(myUserData->key->getShortText())
244  .append('/')
245  .append(QString::number(myUserData->swordFootnote).toUtf8().constData())
246  .append("\">*</span> ");
247 
248  myUserData->swordFootnote++;
249  myUserData->suspendTextPassThru = true;
250  myUserData->inFootnoteTag = true;
251  } else if (tag.isEndTag()) { // end tag
252  // buf += ")</span>";
253  myUserData->suspendTextPassThru = false;
254  myUserData->inFootnoteTag = false;
255  }
256  }
257  } else if (!sword::stricmp(tagName, "scripRef")) { // a scripRef
258  // scrip refs which are embeded in footnotes may not be displayed!
259 
260  if (!myUserData->inFootnoteTag) {
261  if (tag.isEndTag()) {
262  if (myUserData->inscriptRef) { // like "<scripRef passage="John 3:16">See John 3:16</scripRef>"
263  buf.append("</a></span>");
264 
265  myUserData->inscriptRef = false;
266  myUserData->suspendTextPassThru = false;
267  } else { // like "<scripRef>John 3:16</scripRef>"
268  if (CSwordModuleInfo const * const mod =
269  btConfig().getDefaultSwordModuleByType(
270  "standardBible"))
271  {
272  ReferenceManager::ParseOptions options{
273  mod->name(),
274  // current module key:
275  QString::fromUtf8(myUserData->key->getText()),
276  myModule->getLanguage()};
277 
278  //it's ok to split the reference, because to descriptive text is given
279  bool insertSemicolon = false;
280  buf.append("<span class=\"crossreference\">");
281  QStringList const refs(
282  QString::fromUtf8(
283  myUserData->lastTextNode.c_str()).split(";"));
284  QString oldRef; // the previous reference to use as a base for the next refs
285  for (auto const & ref : refs) {
286  if (!oldRef.isEmpty())
287  options.refBase = oldRef; // Use the last ref as a base, e.g. Rom 1,2-3, when the next ref is only 3:3-10
288 
289  // Use the parsed result as the base for the next ref:
290  oldRef = ReferenceManager::parseVerseReference(
291  ref,
292  options);
293 
294  // Prepend a ref divider if we're after the first one
295  if (insertSemicolon)
296  buf.append("; ");
297 
298  buf.append("<a href=\"")
299  .append(
300  ReferenceManager::encodeHyperlink(
301  *mod,
302  oldRef
303  ).toUtf8().constData()
304  )
305  .append("\" crossrefs=\"")
306  .append(oldRef.toUtf8().constData())
307  .append("\">")
308  .append(ref.toUtf8().constData())
309  .append("</a>");
310  insertSemicolon = true;
311  }
312  buf.append("</span>"); //crossref end
313  }
314  myUserData->suspendTextPassThru = false;
315  }
316  } else if (tag.getAttribute("passage") ) {
317  // The passage was given as a parameter value
318  myUserData->inscriptRef = true;
319  myUserData->suspendTextPassThru = false;
320 
321  CSwordModuleInfo * mod = btConfig().getDefaultSwordModuleByType("standardBible");
322  if (! mod)
323  mod = CSwordBackend::instance()->findFirstAvailableModule(CSwordModuleInfo::Bible);
324 
325  if (mod)
326  {
327  ;
328  BT_ASSERT(tag.getAttribute("passage"));
329  QString const completeRef(
330  ReferenceManager::parseVerseReference(
331  QString::fromUtf8(
332  tag.getAttribute("passage")),
333  ReferenceManager::ParseOptions{
334  mod->name(),
335  QString::fromUtf8(
336  myUserData->key->getText()),
337  myModule->getLanguage()}));
338  buf.append("<span class=\"crossreference\">")
339  .append("<a href=\"")
340  .append(
341  ReferenceManager::encodeHyperlink(
342  *mod,
343  completeRef
344  ).toUtf8().constData()
345  )
346  .append("\" crossrefs=\"")
347  .append(completeRef.toUtf8().constData())
348  .append("\">");
349  } else {
350  buf.append("<span><a>");
351  }
352  } else { // We're starting a scripRef like "<scripRef>John 3:16</scripRef>"
353  myUserData->inscriptRef = false;
354  /* Let's stop text from going to output, the text get's
355  added in the -tag handler: */
356  myUserData->suspendTextPassThru = true;
357  }
358  }
359  } else if (!sword::stricmp(tagName, "div")) {
360  if (tag.isEndTag()) {
361  buf.append("</div>");
362  } else if (char const * const tagClass = tag.getAttribute("class")){
363  if (!sword::stricmp(tagClass, "sechead") ) {
364  buf.append("<div class=\"sectiontitle\">");
365  } else if (!sword::stricmp(tagClass, "title")) {
366  buf.append("<div class=\"booktitle\">");
367  }
368  }
369  } else if (!sword::stricmp(tagName, "img") && tag.getAttribute("src")) {
370  const char * value = tag.getAttribute("src");
371 
372  if (value[0] == '/')
373  value++; //strip the first /
374 
375  buf.append("<img src=\"")
376  .append(
377  QUrl::fromLocalFile(
378  QTextCodec::codecForLocale()->toUnicode(
379  myUserData->module->getConfigEntry(
380  "AbsoluteDataPath")
381  ).append('/').append(QString::fromUtf8(value))
382  ).toString().toUtf8().constData())
383  .append("\" />");
384  } else { // Let unknown token pass thru:
385  return sword::ThMLHTML::handleToken(buf, token, userData);
386  }
387  }
388  return true;
389 }
