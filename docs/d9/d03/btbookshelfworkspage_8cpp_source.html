70 inline bool filter(WizardTaskType const taskType,
71  QStringList const & languages,
72  CSwordModuleInfo const * const mInfo)
73 {
74  if (taskType == WizardTaskType::installWorks) {
75  return !CSwordBackend::instance()->findModuleByName(mInfo->name())
76  && languages.contains(mInfo->language()->translatedName());
77  } else if (taskType == WizardTaskType::updateWorks) {
78  using CSMI = CSwordModuleInfo;
79  using CSV = sword::SWVersion const;
80  CSMI const * const installedModule =
81  CSwordBackend::instance()->findModuleByName(mInfo->name());
82  return installedModule
83  && (CSV(installedModule->config(CSMI::ModuleVersion).toLatin1())
84  < CSV(mInfo->config(CSMI::ModuleVersion).toLatin1()));
85  } else {
86  BT_ASSERT(taskType == WizardTaskType::removeWorks);
87  return CSwordBackend::instance()->findModuleByName(mInfo->name());
88  }
89 }

257 void BtBookshelfWorksPage::initializePage() {
258  // Update models:
259  QStringList sources;
260  QStringList languages;
261  if (m_taskType == installWorks) {
262  sources = btWizard().selectedSources();
263  languages = btWizard().selectedLanguages();
264  } else {
265  sources = BtInstallBackend::sourceNameList();
266  }
267 
268  {
269  QSet<QString> addedModuleNames;
270  m_bookshelfModel->clear();
271  m_usedBackends.clear();
272  for (auto const & sourceName : sources) {
273  sword::InstallSource const source =
274  BtInstallBackend::source(sourceName);
275  std::unique_ptr<CSwordBackend const> backend(
276  BtInstallBackend::backend(source));
277  bool backendUsed = false;
278  for (auto * const module : backend->moduleList()) {
279  if (filter(m_taskType, languages, module)) {
280  QString const & moduleName = module->name();
281  if (addedModuleNames.contains(moduleName))
282  continue;
283  addedModuleNames.insert(moduleName);
284  m_bookshelfModel->addModule(module);
285  module->setProperty("installSourceName",
286  QString(source.caption.c_str()));
287  backendUsed = true;
288  }
289  }
290  if (backendUsed)
291  m_usedBackends.emplace_back(std::move(backend));
292  }
293  if (m_taskType != WizardTaskType::installWorks)
294  m_bookshelfView->expandAll();
295 
296  bool const noWorks = (addedModuleNames.count() == 0);
297  m_msgLabel->setVisible(noWorks);
298  m_bookshelfView->setVisible(!noWorks);
299  m_groupingButton->setVisible(!noWorks);
300  m_groupingLabel->setVisible(!noWorks);
301  }
302 
303  // Set grouping:
304  static BtBookshelfTreeModel::Grouping const cat(
305  BtBookshelfTreeModel::GROUP_CATEGORY);
306  static BtBookshelfTreeModel::Grouping const catLang; // No grouping
307  if (languages.count() == 1) {
308  slotGroupingActionTriggered(cat);
309  } else if (languages.count() > 1 && m_groupingOrder == cat) {
310  slotGroupingActionTriggered(catLang);
311  }
312 }

340 static bool installPathIsUsable(QString const & path) {
341  if (path.isEmpty())
342  return false;
343  QDir const dir(path);
344  if (!dir.exists() || !dir.isReadable())
345  return false;
346  return QFileInfo(dir.canonicalPath()).isWritable();
347 }

349 void BtBookshelfWorksPage::slotInitPathCombo() {
350  m_pathCombo->clear();
351  QStringList const targets(BtInstallBackend::targetList());
352  bool haveUsableTargets = false;
353  for (auto const & target : targets) {

356  haveUsableTargets = true;
357  }
358  }
359 
360  m_pathCombo->setVisible(haveUsableTargets);
361  m_pathLabel->setVisible(haveUsableTargets);
362  if (haveUsableTargets) {
363  /* Choose the current value from config but check whether we have so
364  many items: */
365  int const cfgValue = btConfig().value<int>(installPathKey, 0);
366  int const lastIdx = m_pathCombo->count() - 1;
367  m_pathCombo->setCurrentIndex(cfgValue > lastIdx ? lastIdx : cfgValue);
368  }
