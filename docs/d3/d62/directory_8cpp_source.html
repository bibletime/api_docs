278 void removeRecursive(const QString &dir) {
279  //Check for validity of argument
280  if (dir.isEmpty()) return;
281  QDir d(dir);
282  if (!d.exists()) return;
283 
284  //remove all files in this dir
285  d.setFilter( QDir::Files | QDir::Hidden | QDir::NoSymLinks );
286  for (auto const & fileInfo : d.entryInfoList())
287  d.remove(fileInfo.fileName());
288 
289  //remove all subdirs recursively
290  d.setFilter( QDir::Dirs | QDir::NoSymLinks );
291  for (auto const & dirInfo : d.entryInfoList())
292  if (dirInfo.isDir()
293  && dirInfo.fileName() != "."
294  && dirInfo.fileName() != "..")
295  removeRecursive(dirInfo.absoluteFilePath());
296  d.rmdir(dir);
297 }

301 size_t getDirSizeRecursive(QString const & dir) {
302  //Check for validity of argument
303  QDir d(dir);
304  if (!d.exists())
305  return 0u;
306 
307  size_t size = 0u;
308 
309  //First get the size of all files int this folder
310  d.setFilter(QDir::Files);
311  for (auto const & fileInfo : d.entryInfoList()) {
312  BT_ASSERT(fileInfo.size() > 0);
313  size += fileInfo.size();
314  }
315 
316  //Then add the sizes of all subdirectories
317  d.setFilter(QDir::Dirs);
318  for (auto const & dirInfo : d.entryInfoList())
319  if (dirInfo.isDir()
320  && dirInfo.fileName() != "."
321  && dirInfo.fileName() != "..")
322  size += getDirSizeRecursive(dirInfo.absoluteFilePath());
323  return size;
324 }

377 QString getDocFile(QString docName) {
378  auto dir(*cachedDocDir);
379  QString r;
380  if (dir.cd(docName)) {
381  QStringList tryLanguages;
382  tryLanguages.append(QLocale().name());
383  {
384  auto const & localeName = tryLanguages.back();
385  if (localeName.contains('_'))
386  tryLanguages.append(localeName.section('_', 0, 0));
387  }
388  tryLanguages.append("en");
389  auto const tryDoc =
390  [&dir,&tryLanguages](QString const & type,
391  QString const & filename) -> QString
392  {
393  if (dir.cd(type)) {
394  for (auto const & tryLanguage : tryLanguages) {
395  if (dir.cd(tryLanguage)) {
396  if (dir.exists(filename))
397  return dir.absoluteFilePath(filename);
398  dir.cdUp();
399  }
400  }
401  dir.cdUp();
402  }
403  return {};
404  };
405  r = tryDoc("html", "index.html");
406  if (r.isEmpty())
407  r = tryDoc("pdf", docName + ".pdf");
408  }
409  return r;
410 }
