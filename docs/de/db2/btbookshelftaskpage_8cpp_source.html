107int BtBookshelfTaskPage::nextId() const {
108 if (m_installRadioButton->isChecked()) {
109 if (timeToUpdate())
110 return WizardPage::sourcesProgressPage;
111 return WizardPage::sourcesPage;
112 } else if (m_updateRadioButton->isChecked()) {
113 if (timeToUpdate())
