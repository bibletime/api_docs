80 void BtModuleTextModel::reloadModules() {
81 
82  m_moduleInfoList.clear();
83  for (int i = 0; i < m_modules.count(); ++i) {
84  QString moduleName = m_modules.at(i);
85  CSwordModuleInfo* module = CSwordBackend::instance()->findModuleByName(moduleName);
86  m_moduleInfoList.append(module);
87  }
88 
89  beginResetModel();
90  const CSwordModuleInfo* firstModule = m_moduleInfoList.at(0);
91  if (isBible() || isCommentary()) {
92  auto & m = firstModule->module();
93  m.setPosition(sword::TOP);
94  m_firstEntry = m.getIndex();
95  m.setPosition(sword::BOTTOM);
96  m_maxEntries = m.getIndex() - m_firstEntry + 1;
97  } else if(isLexicon()) {
98  m_maxEntries =
99  static_cast<CSwordLexiconModuleInfo const *>(firstModule)
100  ->entries().size();
101  } else if(isBook()) {
102  sword::TreeKeyIdx tk(
103  *static_cast<CSwordBookModuleInfo const *>(firstModule)
104  ->tree());
105  tk.root();
106  tk.firstChild();
107  BT_ASSERT(tk.getOffset() == 4);
108  tk.setPosition(sword::BOTTOM);
109  m_maxEntries = tk.getOffset() / 4;
110  }
111 
112  endResetModel();
113 }

121 QVariant BtModuleTextModel::data(const QModelIndex & index, int role) const {
122 
123  QString text;
124  if (isBible() || isCommentary())
125  text = verseData(index, role);
126  else if (isBook())
127  text = bookData(index, role);
128  else if (isLexicon())
129  text = lexiconData(index, role);
130  else
131  text = "invalid";
132 
133  if (m_textFilter) {
134  text = m_textFilter->processText(text);
135  }
136 
137  if ( ! m_highlightWords.isEmpty()) {
138  QString t = CSwordModuleSearch::highlightSearchedText(text, m_highlightWords);
139  if (m_findState && index.row() == m_findState->index) {
140  // t = highlightFindPreviousNextField(t); now inlined:
141  int from = 0;
142  for (int i = 0; i < m_findState->subIndex; ++i) {
143  int pos = t.indexOf("\"highlightwords\"", from);
144  if (pos == -1)
145  return t;
146  else {
147  from = pos + 1;
148  }
149  }
150  int position = from + 14; // highlightwords = 14, quote was already added
151  t.insert(position, "2");
152  }
153  return QVariant(t);
154  }
155 
156  return QVariant(text);
157 }

211 QString BtModuleTextModel::verseData(const QModelIndex & index, int role) const {
212  int row = index.row();
213  CSwordVerseKey key = indexToVerseKey(row);
214  int verse = key.verse();
215 
216  if (role >= ModuleEntry::TextRole && role <= ModuleEntry::Title9Role) {
217  if (verse == 0)
218  return QString();
219  QString text;
220 
221  QString chapterTitle;
222  if (verse == 1)
223  chapterTitle = key.bookName() + " " + QString::number(key.chapter());
224 
225  BtConstModuleList modules;
226  if ( role == ModuleEntry::TextRole) {
227  modules = m_moduleInfoList;
228  } else {
229  int column = getColumnFromRole(role);
230  CSwordModuleInfo const * module;
231  if ((column + 1) > m_moduleInfoList.count())
232  module = m_moduleInfoList.at(0);
233  else
234  module = m_moduleInfoList.at(column);
235  modules.append(module);
236  CSwordVerseKey mKey(module);
237  mKey.setKey(key.key());
238 
239  // Title only for verse 1 of Personal commentary
240  if (role >= ModuleEntry::Title0Role && role <= ModuleEntry::Title9Role) {
241  if (module->isWritable() && verse == 1)
242  return "<center><h3>" + chapterTitle + "</h3></center>";
243  return "";
244  }
245 
246  // Personal commentary
247  if (module->isWritable()) {
248  QString text;
249  QString rawText = mKey.rawText();
250  if (rawText.isEmpty())
251  rawText = "<span style=\"color:gray\"><small>" + tr("Click to edit") + "</small></span>";
252  text += QString::number(verse) + " " + rawText;
253 
254  text = ColorManager::instance().replaceColors(text);
255  return CSwordModuleSearch::highlightSearchedText(text, m_highlightWords);
256  }
257  }
258 
259  text += Rendering::CEntryDisplay().textKeyRendering(modules,
260  key.key(), m_displayOptions, m_filterOptions,
261  m_displayOptions.verseNumbers ?
262  Rendering::CTextRendering::KeyTreeItem::Settings::SimpleKey :
263  Rendering::CTextRendering::KeyTreeItem::Settings::NoKey);
264 
265  text.replace("#CHAPTERTITLE#", chapterTitle);
266  text = ColorManager::instance().replaceColors(text);
267  return text;
268  }
269  return QString();
270 }
