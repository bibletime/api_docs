35 QString getColorByPattern(
36  std::map<QString, std::map<QString, QString> > const & maps,
37  QString const & pattern,
38  QString const & style)
39 {
40  QString activeTemplate;
41  if (style.isEmpty())
42  activeTemplate = CDisplayTemplateMgr::activeTemplateName();
43  auto const mapIt(maps.find(activeTemplate));
44  BT_ASSERT(mapIt != maps.end());
45  auto const valueIt(mapIt->second.find(pattern));
46  BT_ASSERT(valueIt != mapIt->second.end());
47  BT_ASSERT(!valueIt->second.isEmpty());
48  return valueIt->second;

61 void ColorManager::loadColorMaps() {
62  namespace DU = util::directory;
63  QDir::Filters const readableFileFilter(QDir::Files | QDir::Readable);
64  QStringList const cssFilter("*.css");
65 
66  auto const loadColorMap =
67  [this](QString const & filePath) {
68  QFileInfo const cssInfo(filePath);
69  static QString const cMapPathTemplate("%1/%2.cmap");
70  auto cMapPath(cMapPathTemplate.arg(cssInfo.path())
71  .arg(cssInfo.completeBaseName()));
72  QFileInfo const cMapInfo(cMapPath);
73  auto fileName(cssInfo.fileName());
74 
75  // Start with default color map:
76  std::map<QString, QString> colorMap;
77  auto const p(qApp->palette());
78  if (darkMode()) {
79  colorMap.emplace("FOREGROUND_COLOR",
80  p.color(QPalette::WindowText).name());
81  colorMap.emplace("BACKGROUND_COLOR",
82  p.color(QPalette::Base).name());
83  colorMap.emplace("HIGHLIGHT",
84  QColor("#ffff00").name());
85  colorMap.emplace("BACKGROUND_HIGHLIGHT",
86  QColor("#444466").name());
87  colorMap.emplace("CROSSREF_COLOR",
88  QColor("#aac2ff").name());
89  colorMap.emplace("JESUS_WORDS_COLOR",
90  QColor("#ff0000").name());
91  } else {
92  colorMap.emplace("FOREGROUND_COLOR",
93  p.color(QPalette::WindowText).name());
94  colorMap.emplace("BACKGROUND_COLOR",
95  p.color(QPalette::Base).name());
96  colorMap.emplace("HIGHLIGHT",
97  QColor("#ffff00").name());
98  colorMap.emplace("BACKGROUND_HIGHLIGHT",
99  QColor("#ddddff").name());
100  colorMap.emplace("CROSSREF_COLOR",
101  QColor("#1414ff").name());
102  colorMap.emplace("JESUS_WORDS_COLOR",
103  QColor("#ff0000").name());
104  }
105 
106  if (cMapInfo.exists()) {
107  QSettings cMapSettings(cMapPath, QSettings::IniFormat);
108  static QString const dark("dark");
109  static QString const light("light");
110  cMapSettings.beginGroup(darkMode() ? dark : light);
111  for (auto const & colorKey : cMapSettings.childKeys())
112  colorMap[colorKey] = cMapSettings.value(colorKey).toString();
113  }
114  m_colorMaps[std::move(fileName)] = std::move(colorMap);
115  };
116 
117  // Load global app stylesheets
118  auto const & td = DU::getDisplayTemplatesDir();
119  for (auto const & file : td.entryList(cssFilter, readableFileFilter))
120  loadColorMap(td.canonicalPath() + "/" + file);
121 
122  // Load user app stylesheets
123  auto const & utd = DU::getUserDisplayTemplatesDir();
124  for (auto const & file : utd.entryList(cssFilter, readableFileFilter))
125  loadColorMap(utd.canonicalPath() + "/" + file);
126 }
