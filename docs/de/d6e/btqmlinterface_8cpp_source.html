138 int BtQmlInterface::getCurrentModelIndex() const {
139  if (m_swordKey == nullptr)
140  return 0;
141  if (moduleIsBibleOrCommentary(module())) {
142  CSwordVerseKey* verseKey = dynamic_cast<CSwordVerseKey*>(m_swordKey);
143  int index = m_moduleTextModel->verseKeyToIndex(*verseKey);
144  return index;
145  }
146  else if (moduleIsBook(module())) {
147  const CSwordBookModuleInfo *m = qobject_cast<const CSwordBookModuleInfo*>(module());
148  CSwordTreeKey key(m->tree(), m);
149  QString keyName = m_swordKey->key();
150  key.setKey(keyName);
151  CSwordTreeKey p(key);
152  p.positionToRoot();
153  if(p != key)
154  return static_cast<int>(key.offset() / 4u); /// \todo Check range!
155  }
156  else if (moduleIsLexicon(module())){
157  const CSwordLexiconModuleInfo *li =
158  qobject_cast<const CSwordLexiconModuleInfo*>(m_swordKey->module());
159  int index = li->entries().indexOf(m_swordKey->key());
160  return index;
161  }

253 QString BtQmlInterface::getReferenceFromUrl(const QString& url) {
254  QString reference;
255 
256 
257  QRegExp rx("sword://(bible|lexicon)/(.*)/(.*)(\\|\\|)", Qt::CaseInsensitive);
258  rx.setMinimal(false);
259  int pos1 = rx.indexIn(url);
260  if (pos1 > -1) {
261  reference = "href=sword://" + rx.cap(1) + "/" + rx.cap(2) + "/" + rx.cap(3);
262  } else {
263  QRegExp rx0("sword://(bible|lexicon)/(.*)/(.*)", Qt::CaseInsensitive);
264  rx0.setMinimal(false);
265  int pos1 = rx0.indexIn(url);
266  if (pos1 > -1) {
267  reference = "href=sword://" + rx0.cap(1) + "/" + rx0.cap(2) + "/" + rx0.cap(3);
268 
269  } else {
270  QRegExp rx1("sword://footnote/(.*)=(.*)", Qt::CaseInsensitive);
271  rx1.setMinimal(false);
272  int pos1 = rx1.indexIn(url);
273  if (pos1 > -1) {
274  reference = "note=" + rx1.cap(1);
275 
276  } else {
277  QRegExp rx2("sword://lemmamorph/(.*)=(.*)/(.*)", Qt::CaseInsensitive);
278  rx2.setMinimal(false);
279  int pos1 = rx2.indexIn(url);
280  if (pos1 > -1) {
281  reference = rx2.cap(1) + "=" + rx2.cap(2);
282  }
283  }
284  }
285  }

607 void BtQmlInterface::getNextMatchingItem(int startIndex) {
608  int num = countHighlightsInItem(startIndex);
609  if (num > m_findState->subIndex) { // Found within startIndex item
610  m_findState->index = startIndex;
611  ++m_findState->subIndex;
612  return;
613  }
614 
615  if (startIndex >= m_moduleTextModel->rowCount())
616  return;
617 
618  int index = startIndex+1;
619  for (int i = 0; i < 1000; ++i) {
620  int num = countHighlightsInItem(index);
621  if (num > 0 ) {
622  m_findState->index = index;
623  m_findState->subIndex = 1;
624  return;
625  }
626  ++index;
627  }
628  return;
629 }

631 void BtQmlInterface::getPreviousMatchingItem(int startIndex) {
632  int num = countHighlightsInItem(startIndex);
633  if (num > 0 && m_findState->subIndex == 0) {
634  // Found within startIndex item
635  m_findState->index = startIndex;
636  m_findState->subIndex = 1;
637  return;
638  }
639 
640  if (startIndex <= 0)
641  return;
642 
643  int index = startIndex;
644  if (m_findState->subIndex == 0)
645  --index;
646  for (int i = 0; i < 1000; ++i) {
647  int num = countHighlightsInItem(index);
648  if (num > 0 ) {
649  m_findState->index = index;
650  if (m_findState->subIndex == 0)
651  m_findState->subIndex = num;
652  else
653  --m_findState->subIndex;
654  if (m_findState->subIndex != 0)
655  return;
656  }
657  --index;
658  }
659  return;
660 }
